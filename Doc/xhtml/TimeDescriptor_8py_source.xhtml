51  def __eq__(self,other):
52  """overloads ==
53  @param other instance of class TimeDescriptor
54  @return boolean whether the other descriptor is the same.

69  def Times(self,unit=None):
70  """returns list of times
71  @param unit (optional) string of units for time values.
72  @return list of times corresponding to self.
73  @note valid time units are:
74  - None - return simply the list of time values.
75  - 'ps' - return the time values divided by 1e-12.
76  - 'ns' - return the time values divided by 1e-9.
77  - 'us' - return the time values divided by 1e-6.
78  - 'ms' - return the time values divided by 1e-3.
79  """
80  if unit==None:
81  return [self[k] for k in range(len(self))]
82  elif isinstance(unit,float):
83  return [self[k]/unit for k in range(len(self))]
84  elif unit=='ps':
85  return [self[k]/1.e-12 for k in range(len(self))]
86  elif unit=='ns':
87  return [self[k]/1.e-9 for k in range(len(self))]
88  elif unit =='us':

92  def ApplyFilter(self,F):
93  """calculates effect of filter
94  calculates the new time descriptor of a waveform filtered by a filter
95  with the filter descriptor provided.
96  @param F instance of class FilterDescriptor
97  @return instance of class TimeDescriptor containing the time descriptor
98  affected by the filter.
99  @note does not affect self.

117  def __div__(self,other):
118  return self.__truediv__(other)
119  def __truediv__(self,other):
120  """overloads /
121  This is an abstraction and is polymorphic.
122  if other is a filter descriptor, it assumes self is an outputwf such that
123  inputwf*other=outputwf and it calculates the inputwf as outputwf*other^-1.
124  This is the same as finding the input waveform that if filtered by other would
125  produce self.
126  if other is an input wf time descriptor, it assume that self is an output wf such that
127  inputwf*filter=outputwf and it calculates the filter as inputwf^-1*outputwf.
128  This is the same as finding the filter that, when applied to other, would produce self.
129  @param other instance of class TimeDescriptor or FilterDescriptor.
130  @return instance of class TimeDescriptor or FilterDescriptor depending on the situations
131  explained.
132  """
133  if isinstance(other,FilterDescriptor):
134  return TimeDescriptor(
135  HorOffset=self.H+other.U/self.Fs*(other.D-other.S),
136  NumPts=self.K/other.U+other.S,
137  SampleRate=self.Fs/other.U)
138  elif isinstance(other,TimeDescriptor):
139  UpsampleFactor=self.Fs/other.Fs
140  return FilterDescriptor(
141  UpsampleFactor,
142  DelaySamples=other.K-self.K/UpsampleFactor-
143  (self.H-other.H)*other.Fs,
144  StartupSamples=other.K-self.K/UpsampleFactor)
145  def DelayBy(self,D):
146  """delays by time

153  """corresponding frequency list

164  """intersection of two waveforms
