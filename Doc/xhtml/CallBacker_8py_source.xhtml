22 class CallBacker(object):
23  """CallBacker
24  To use callbacks, a class is derived from CallBacker.
25  Either the constructor of the derived class or some other mechanism should
26  provide means for supplying the callback function, which is installed in a
27  call to __init__.
28  The derived class, during some long operation, then calls the callback function
29  periodically. The callback function then has the opportunity to report its
30  progress. Returning False in the callback function should cause the operation
31  to abort.
32  Here's a mini example derived class using CallBacker:
33  \code{.py}
34  class ProcessingClass(Callbacker):
35  def __init__(self,callback=None):
36  ...
37  Callbacker.__init__(callback)
38  ...
39  def LongProcessingFunction(self):
40  for m in range(100):
41  ...
42  do some calculating.
43  ...
44  if not self.CallBack(m):
45  process was aborted
46  return
47  \endcode
48  """
49  def __init__(self,callback=None):
50  """Constructor
51  @param callback (optional, defaults to None) a callback function to call
52  The callback function should take a number as an argument, which represents
53  the progress."""
54  self.InstallCallback(callback)
55  def HasACallBack(self):
56  return not self.callback is None
57  def CallBack(self,progressPercent):
58  """This function is called periodically, which in turn calls any installed
59  callback function.
60  @param progressPercent the progress in percent
61  """
62  if not self.callback is None:
63  return self.callback(progressPercent)
64  else:
65  return True
66  def InstallCallback(self,callback=None):
67  """This is an alternate way to supply the callback, if it cannot be installed
68  during initialization of the derived class.
69  @param callback (optional, defaults to None) a callback function to call
70  The callback function should take a number as an argument, which represents
71  the progress.
72  """
73  self.callback = callback
74  def RemoveCallback(self):
75  """Removes any callback previously installed."""
76  self.InstallCallback()
