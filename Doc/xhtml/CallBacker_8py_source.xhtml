21 class CallBacker(object):
22  """CallBacker
23 
24  To use callbacks, a class is derived from CallBacker.
25 
26  Either the constructor of the derived class or some other mechanism should
27  provide means for supplying the callback function, which is installed in a
28  call to __init__.
29 
30  The derived class, during some long operation, then calls the callback function
31  periodically. The callback function then has the opportunity to report its
32  progress. Returning False in the callback function should cause the operation
33  to abort.
34 
35  Here's a mini example derived class using CallBacker:
36 
37  \code{.py}
38  class ProcessingClass(Callbacker):
39  def __init__(self,callback=None):
40  ...
41  Callbacker.__init__(callback)
42  ...
43  def LongProcessingFunction(self):
44  for m in range(100):
45  ...
46  do some calculating.
47  ...
48  if not self.CallBack(m):
49  process was aborted
50  return
51  \endcode
52  """
53  def __init__(self,callback=None):
54  """Constructor
55  @param callback (optional, defaults to None) a callback function to call
56  The callback function should take a number as an argument, which represents
57  the progress."""
58  self.InstallCallback(callback)
59  def HasACallBack(self):
60  return not self.callback is None
61  def CallBack(self,progressPercent):
62  """This function is called periodically, which in turn calls any installed
63  callback function.
64  @param progressPercent the progress in percent
65  """
66  if not self.callback is None:
67  return self.callback(progressPercent)
68  else:
69  return True
70  def InstallCallback(self,callback=None):
71  """This is an alternate way to supply the callback, if it cannot be installed
72  during initialization of the derived class.
73  @param callback (optional, defaults to None) a callback function to call
74  The callback function should take a number as an argument, which represents
75  the progress.
76  """
77  self.callback = callback
78  def RemoveCallback(self):
79  """Removes any callback previously installed."""
80  self.InstallCallback()
