30 class OpticalCalculator(dict):
31  """Optical Measurements Calculator
32  This class enables optical measurements attributed to optical modulators.
33  The simplest usage of the optical calculator is like this:
34 
35  params = OpticalCalculator('PindBm',7.17,'IILdB',2.0,'ILdB',2.87,'ERdB',4.63).Result()
36 
37  This results in params containing a dictionary of calculated results like this:
38 
39  params = {'PindBm': 7.17, 'PinW': 0.005212, 'ILdB': 2.87, 'IL': 1.936422, 'ERdB': 4.63, 'ER': 2.904023, 'PHdBm': 4.3,
40  'PHW': 0.002692, 'PLdBm': -0.33, 'PLW': 0.000927, 'OMAdBm': 2.466721, 'OMAW': 0.001765, 'TPdB': 4.703279,
41  'TP': 2.953438, 'PmaxdBm': 5.17, 'PmaxW': 0.003289, 'PavgdBm': 2.574823, 'PavgW': 0.001809, 'IILdB': 2.0,
42  'IIL': 1.584893, 'LossdB': 4.595177, 'Loss': 2.88083, 'MLdB': 2.595177, 'ML': 1.817681}
43 
44  These parameters are all of the linear and logarithmic solutions for the optical calculations. The definitions of
45  these parameters can be seen in the image obtained from OpticalCalculator.Picture().
46 
47  @image{inline} html OpticalCalculations.png "optical parameter definitions"

51  def __init__(self,*kwargs):
52  """Constructor for an optical calculator
53  @param *kwargs optional tuple containing keyword/value pairs for optical parameters to add at construction
54  If the constructor is called with no arguments, kwargs = (), an empty tuple. Otherwise the keyword and values
55  are added to the optical calculator using the AddParam function.
56  @see AddParam
57  """
58  dict.__init__(self,{})
59  self.validunit = {}
60  for base_name in self.param_dict.keys():
61  param=self.param_dict[base_name]
62  self.validunit[param['log']['name']]={'unit':param['log']['unit']}
63  self.validunit[param['lin']['name']]={'unit':param['lin']['unit']}
64  if kwargs != ():
65  if len(kwargs)//2*2 != len(kwargs):
66  raise IndexError('optical arguments must come in key/value pairs')
67  for kv in range(len(kwargs)//2):
68  if not self.AddParam(kwargs[kv*2],kwargs[kv*2+1],True):
69  raise ValueError('optical calculator does not accept : '+str(kwargs[kv*2])+' = '+str(kwargs[kv*2+1]))

71  def AddParam(self,name,value=None,fixed=False,calculate=True):
72  """Adds a parameter to the optical calculator
73  @param name string name of a parameter. It must be one of the recognized parameter names in the
74  calculator, otherwise it will throw an exception.
75  @param value float value to assign to the named parameter
76  @param fixed bool (optional, defaults to False) whether the parameter is fixed. If it is fixed, it checks for overconstrained
77  parameter addition
78  @param calculate bool (optional, defaults to True) whether to try to compute all of the other parameters upon addition.
79  @return bool True if the parameter could be added, False if the parameter could not be added, either because the keyword was
80  not identified or because the parameter calculation has already been made.
81  """
82  if calculate: self.CalculateAll()
83  if name in self.validunit.keys():
84  if name in self.keys():
85  if self[name]['value'] != None:
86  return False
87  self[name]=OpticalParam(name,fixed,value,unit=self.validunit[name]['unit'])
88  if calculate: self.CalculateAll()
89  return True
90  return False

234  def CalcDiff(self,name,params,only=False):
235  """Calculates a parameter value based on the difference between two other parameters
236  @param name string name of parameter to calculate
237  @param params list of two string names of other parameters
238  @param only bool (optional, defaults to False) whether to try to compute other permutations
239  if the named parameter is a and the params are b,c then the permutations are:
240  * a=b-c
241  * b=a+c
242  * c=b-a
243  If only is False, only a=b-c is computed.
244  """
245  if self.NeedsCalculation(name):
246  if self.HasValues(params):
247  self.SetValue(name,self.GetValue(params[0])-self.GetValue(params[1]))
248  return True
249  elif not only and self.CalcSum(params[0],[name,params[1]],True):
250  return True
251  elif not only and self.CalcDiff(params[1],[params[0],name],True):
252  return True
253  return False

256  """Calculates a parameter value based on the sum of two other parameters
257  @param name string name of parameter to calculate
258  @param params list of two string names of other parameters
259  @param only bool (optional, defaults to False) whether to try to compute other permutations
260  if the named parameter is a and the params are b,c then the permutations are:
261  * a=b+c
262  * b=a-c
263  * c=a-b
264  If only is False, only a=b+c is computed.

271  return True

276  def CalculateAll(self):
277  """Calculates all possible values that can be calculated based on the current population of the dictionary
278  """
279  self.LindBdBm()
280  Added = True
281  while Added:
282  Added=False

284  Added=True
285  elif self.CalcDiff('OMAW',['PHW','PLW']):

288  Added=True
289  elif self.CalcDiff('ERdB',['PHdBm','PLdBm']):
290  Added=True
291  elif self.CalcDiff('PmaxdBm',['PindBm','IILdB']):
292  Added=True
293  elif self.CalcEq('PavgW',['PHW','PLW'],'(self.GetValue(params[0])+self.GetValue(params[1]))/2.0'):
294  Added=True
295  elif self.CalcEq('PHW',['PavgW','PLW'],'2.*self.GetValue(params[0])-self.GetValue(params[1])'):
296  Added=True
297  elif self.CalcEq('PLW',['PavgW','PHW'],'2.*self.GetValue(params[0])-self.GetValue(params[1])'):
298  Added=True
299  elif self.CalcDiff('LossdB',['PindBm','PavgdBm']):
300  Added=True
301  elif self.CalcSum('PmaxdBm',['PavgdBm','MLdB']):
302  Added=True
303  if Added:
304  self.LindBdBm()
