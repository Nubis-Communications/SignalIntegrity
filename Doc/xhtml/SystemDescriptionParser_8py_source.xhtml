77  def _ProcessLine(self,line,exclusionList):
78  """processes the lines of a netlist
79  Lines that can be processed at this level are processed and lines that
80  are unknown are place in a list of unknown lines for upstream processing. This
81  enables derived classes to benefit from what this class knows how to process and
82  to simply add specific functionality. As a simple example, a derived simulator class
83  needs to add output probes, and this simple system description class knows nothing of
84  this.
85  netlist lines that are handled at this level are:
86  - 'device' - addition of devices.
87  - 'connect' - handles device connections.
88  - 'port' - adds system ports for s-parameter calculation.
89  @param exclusionList list of strings representing commands to exclude. These are either
90  commands that are removed from the functionality (i.e. it would not be write to add a system
91  port in a simulation), or commands that are withheld until further processing at a later time
92  to enforce the order that commands are handled in. For example, we like the netlist to be agnostic
93  about the order of devices listed and device connections, but we cannot connect device ports until
94  a device has been declared.
95  """

98  from SignalIntegrity.Lib.Helpers.LineSplitter import LineSplitter

103  elif lineList[0] in exclusionList: self.m_ul.append(line)

111  if not dev.m_spf is None:

113  self.m_spcl.append([lineList[2]]+argList)
