169 def to_trc(wf,fname):
170  """Save the waveform in lecroy trc format
171  @param wf Instance of class Waveform containing the waveform to store.
172  @param fname String name of the filename to save to. Should have a .trc extension
173  """
174  try:
175  filename, file_extension = os.path.splitext(fname)
176  extensionCorrect=True
177  if file_extension=='' or file_extension is None:
178  fname=fname+'.trc'
179  else:
180  file_extension=file_extension.lower()
181  if file_extension != '.trc':
182  extensionCorrect=False
183  if not extensionCorrect:
184  raise SignalIntegrityExceptionWaveformFile('incorrect extension LeCroy trace file name in '+fname+'. Should be .trc.')
185  except:
186  raise SignalIntegrityExceptionWaveformFile('incorrect extension in LeCroy trace file name in '+fname+'. Should be .trc.')
187 
188  try:
189  # setup the descriptor and populate it with the default values
190  desc = np.empty((1,),dtype=WFM_DESC)
191  for field in default_descriptor_values:
192  desc[field] = default_descriptor_values[field]
193 
194  # TODO: add option to store the raw samples directly rather than re-scaling to get maximum resolution (this might be more important for the ADC tester)
195  # force the waveform into the right format
196  x = np.array(wf.Values())
197  vert_scale = (x.max() - x.min())/60000
198  vert_offset = x.min() + (x.max() - x.min())/2
199 
200  # modify the descriptor where needed
201  desc["WAVE_ARRAY_COUNT"]= len(x), # exact number of samples in the waveform
202  desc["NOM_SUBARRAY_COUNT"]= 1, # the number of segments in the waveform
203  desc["PNTS_PER_SCREEN"]= len(x), # set to 2 less than the total number of points?
204  desc["HORIZ_OFFSET"]= wf.td.H, # time of the first point at the left (first one or first one on screen?)
205  desc["HORIZ_INTERVAL"]= 1/wf.td.Fs, # 1/sample rate
206  desc["LAST_VALID_PNT"]= len(x)-1, # WAVE_ARRAY_COUNT - 1
207  desc["WAVE_ARRAY_1"]= 2*len(x), # 2 * WAVE_ARRAY_COUNT if the data type is word
208  desc["MIN_VALUE"]= -32000., # the data is 16 bits so this is the minimum value
209  desc["MAX_VALUE"]= 31744., # the data is 16 bits so this is the minimum value
210  desc["ACQ_VERT_OFFSET"]= -vert_offset, # the vertical offset of the channel
211  desc["PIXEL_OFFSET"]= wf.td.H, # not sure
212  desc["VERTICAL_OFFSET"]= -vert_offset, # vertical offset but different than ACQ_VERT_OFFSET?
213  desc["VERTICAL_GAIN"] = vert_scale, # volts per 16 bit code
214  desc["FIXED_VERT_GAIN"] = 15
215 
216  # reload these to acquire the quantization effect of storing them in the descriptor
217  vert_scale = desc["VERTICAL_GAIN"][0]
218  vert_offset = -desc["VERTICAL_OFFSET"][0]
219 
220  # it seems that if vert_scale becomes zero, it's safe to just
221  # skip dividing by it.
222  if vert_scale==0.0:
223  x_int = (x - vert_offset).astype(dtype="<i2")
224  else:
225  x_int = ((x - vert_offset)/vert_scale).astype(dtype="<i2")
226 
227  # Write the values to file
228  total_length = 2*len(x) + 346
229 
230  with open(fname, "wb") as f:
231  if sys.version_info.major > 2:
232  f.write(bytes('#9{:09d}'.format(total_length), encoding='utf-8'))
233  else:
234  f.write(b'#9{:09d}'.format(total_length))
235  f.write(desc)
236  # f.write('\00')
237  f.write(x_int.tobytes())
238  # f.write('\00')
239  except:
240  raise SignalIntegrityExceptionWaveformFile('LeCroy trace file could not be written: '+fname)

242 def from_trc(filename):
243  """Read a waveform in lecroy trc format
244  @param filename String name of the filename to read. Should have a .trc extension
245  @return a waveform in SignalIntegrity format
246  """
247  try:
248  fname, file_extension = os.path.splitext(filename)
249  extensionCorrect=True
250  if file_extension=='' or file_extension is None:
251  filename=filename+'.trc'
252  else:
253  file_extension=file_extension.lower()
254  if file_extension != '.trc':
255  extensionCorrect=False
256  if not extensionCorrect:
257  raise SignalIntegrityExceptionWaveformFile('incorrect extension LeCroy trace file name in '+filename+'. Should be .trc.')
258  except:
259  raise SignalIntegrityExceptionWaveformFile('incorrect extension in LeCroy trace file name in '+filename+'. Should be .trc.')
260  try:
261  with open(filename, "rb") as f:
262  header=f.read(11).decode()
263  start=header[0:2]
264  if start != '#9':
265  raise SignalIntegrityExceptionWaveformFile('Incorrect LeCroy trace file header in: '+filename)
266  count=int(header[2:11])
267  descData=np.frombuffer(f.read(346),np.dtype(WFM_DESC))
268  wfBuffer=np.frombuffer(f.read(count-346),'int16')
269  if descData["NOM_SUBARRAY_COUNT"] != 1:
270  raise SignalIntegrityExceptionWaveformFile('Cannot read multi-segment waveform in: '+filename)
271  vertScale=descData['VERTICAL_GAIN'][0]
272  vertOffset=-descData['VERTICAL_OFFSET'][0]
273  horizontalOffset=descData['HORIZ_OFFSET'][0]
274  sampleRate=1./descData['HORIZ_INTERVAL'][0]
275  Exp=10**round(math.log10(sampleRate),6)
276  sampleRate=sampleRate/Exp
277  sampleRate=round(sampleRate,6)
278  sampleRate=sampleRate*Exp
279  numPoints=descData['WAVE_ARRAY_COUNT'][0]
280  from SignalIntegrity.Lib.TimeDomain.Waveform.Waveform import Waveform
281  from SignalIntegrity.Lib.TimeDomain.Waveform.TimeDescriptor import TimeDescriptor
282  wf=Waveform(TimeDescriptor(horizontalOffset,numPoints,sampleRate),
283  [v*vertScale+vertOffset for v in wfBuffer])
284  return wf
285  except:
286  raise SignalIntegrityExceptionWaveformFile('LeCroy trace file could not be read: '+filename)
