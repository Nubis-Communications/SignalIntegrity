21 def Tee(P=3):
22  """symbolic Tee
23  @param P (optional) integer number of ports (defaults to three)\n
24  @return list of list of string s-parameter matrix
25  containing LaTeX or ASCII strings for each element.
26  @note strings can be any valid LaTeX
27  @note this is the symbolic equivalent of SignalIntegrity.Lib.Devices.Tee.Tee
28  """
29  D=str(P)
30  DiagEle='-\\frac{'+str(-(2-P))+'}{'+D+'}'
31  OffDiagEle='\\frac{2}{'+D+'}'
32  M=[]
33  for r in range(P):
34  row=[]
35  for c in range(P):
36  if r==c:
37  ele=DiagEle
38  else:
39  ele=OffDiagEle
40  row.append(ele)
41  M.append(row)
42  return M
