63 def TLineTwoPort(rho,gamma):
64  """symbolic single-ended two-port transmission line
65  @param rho string reflection coefficient
66  @param gamma string gamma
67  @return list of list of string s-parameter matrix
68  containing LaTeX or ASCII strings for each element.
69  @note strings can be any valid LaTeX
70  @note use SignalIntegrity.Symbolic.TLine.TLineRho if you want to enter the string
71  representation of rho based on the characteristic impedance.
72  @note use SignalIntegrity.Symbolic.TLine.TLineGamma if you want to enter the string
73  representation of gamma based on a time delay
74  @note this is the symbolic version of SignalIntegrity.Devices.TLineTwoPort.TLineTwoPort except that by
75  default it's base on rho for symbolic simplicity (Zc complicates the look of things).
76  """
77  L='e^{'+gamma+'}'
78  L2='e^{2 \\cdot '+gamma+' } '
79  D='1 - \\left[ '+rho+' \\right]^2 \\cdot '+L2
80  S1=' \\frac{ '+rho+' \\cdot \\left( 1 - '+L2+' \\right) }{ '+D+' }'
81  S2=' \\frac{ '+L+' \\cdot \\left( 1 - \\left[ '+rho+' \\right] ^2 \\right) }{ '+D+' } '
82  return [[S1,S2],[S2,S1]]

98  Y='e^{'+gamma+' } '
99  Y2='e^{2 \\cdot '+gamma+' } '
100  D='2 \\cdot \\left( 1 - '+Y2+' \\cdot \\left[ '+rho+' \\right]^2 \\right)'
101  S1= '\\frac{ 1 - '+Y2+' \\cdot \\left[ '+rho+' \\right]^2 + '+rho+' \\cdot \\left( 1 - '+Y2+' \\right) }{ '+D+' } '

103  S3= '\\frac{ 1 - '+Y2+' \\cdot \\left[ '+rho+' \\right]^2 - '+rho+' \\cdot \\left( 1 - '+Y2+' \\right) }{ '+D+' } '
104  return [[S1,S2,S3,'-'+S2],
