62 def TLineTwoPort(rho,gamma):
63  """symbolic single-ended two-port transmission line
64  @param rho string reflection coefficient
65  @param gamma string gamma
66  @return list of list of string s-parameter matrix
67  containing LaTeX or ASCII strings for each element.
68  @note strings can be any valid LaTeX
69  @note use SignalIntegrity.Symbolic.TLine.TLineRho if you want to enter the string
70  representation of rho based on the characteristic impedance.
71  @note use SignalIntegrity.Symbolic.TLine.TLineGamma if you want to enter the string
72  representation of gamma based on a time delay
73  @note this is the symbolic version of SignalIntegrity.Lib.Devices.TLineTwoPort.TLineTwoPort except that by
74  default it's base on rho for symbolic simplicity (Zc complicates the look of things).
75  """
76  L='e^{'+gamma+'}'
77  L2='e^{2 \\cdot '+gamma+' } '
78  D='1 - \\left[ '+rho+' \\right]^2 \\cdot '+L2
79  S1=' '+lfrac(' '+rho+' \\cdot \\left( 1 - '+L2+' \\right) ',' '+D+' ')
80  S2=' '+lfrac(' '+L+' \\cdot \\left( 1 - \\left[ '+rho+' \\right] ^2 \\right) ',' '+D+' ')+' '
81  return [[S1,S2],[S2,S1]]

83 def TLineFourPort(rho,gamma):
84  """symbolic single-ended four-port transmission line
85  @param rho string reflection coefficient
86  @param gamma string gamma
87  @return list of list of string s-parameter matrix
88  containing LaTeX or ASCII strings for each element.
89  @note strings can be any valid LaTeX
90  @note use SignalIntegrity.Symbolic.TLine.TLineRho if you want to enter the string
91  representation of rho based on the characteristic impedance.
92  @note use SignalIntegrity.Symbolic.TLine.TLineGamma if you want to enter the string
93  representation of gamma based on a time delay
94  @note this is the symbolic version of SignalIntegrity.Lib.Devices.TLineFourPort.TLineFourPort except that by
95  default it's base on rho for symbolic simplicity (Zc complicates the look of things).
96  """
97  Y='e^{'+gamma+' } '
98  Y2='e^{2 \\cdot '+gamma+' } '
99  D='2 \\cdot \\left( 1 - '+Y2+' \\cdot \\left[ '+rho+' \\right]^2 \\right)'
100  S1= lfrac(' 1 - '+Y2+' \\cdot \\left[ '+rho+' \\right]^2 + '+rho+' \\cdot \\left( 1 - '+Y2+' \\right) ',' '+D+' ')+' '
101  S2= lfrac(' \\left( 1 - \\left[ '+rho+' \\right]^2 \\right) \\cdot '+Y+' ',' '+D+' ')+' '
102  S3= lfrac(' 1 - '+Y2+' \\cdot \\left[ '+rho+' \\right]^2 - '+rho+' \\cdot \\left( 1 - '+Y2+' \\right) ',' '+D+' ')+' '
103  return [[S1,S2,S3,'-'+S2],
104  [S2,S1,'-'+S2,S3],
105  [S3,'-'+S2,S1,S2],
106  ['-'+S2,S3,S2,S1]]
