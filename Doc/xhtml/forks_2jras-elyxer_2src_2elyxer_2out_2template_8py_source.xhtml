251  "Replace all variables in a line."
252  result = ''
253  pos = TextPosition(line)
254  while not pos.finished():

261  def getvalue(self, pos):
262  "Get the value of the variable at the given position."

266  Trace.error('Template variable ' + key + ' not found')
267  else:

270  Trace.error('Weird template format in ' + line)
