26  """assigns arguments
27  @param args list of arguments as name,value pairs
28  The arguments are stored as a dictionary of values corresponding to the keyword names
29  """
30  self.m_vars=dict()
31  if args is None:
32  self.m_args=[]
33  else:
34  args=LineSplitter(args)
35  self.m_args = dict([('$'+args[i]+'$',args[i+1])
36  for i in range(0,len(args),2)])
37  def ReplaceArgs(self,lineList):
38  """replaces arguments in a line list with the actual values
39  @param lineList list of tokens in a line to process
40  If any token is surrounded by '$' and it's name is in the variable dictionary, then
41  the token is replaced with the value in the variable dictionary.
42  """
43  replacedOne=False
44  for i in range(len(lineList)):
45  if lineList[i] in self.m_vars:
46  replacedOne=True
47  lineList[i] = self.m_vars[lineList[i]]
48  if replacedOne:
49  lineList=';'.join(lineList).split(';')
50  return lineList
51  def ProcessVariables(self,lineList):
52  """processes lines insofar as arguments are concerned.
53  @param lineList list of tokens in a line to process
54  If the first token is 'var', then the keyword,value pairs are placed in
55  the variable dictionary
56  """
57  if lineList[0] == 'var':
58  variables=dict([(lineList[i*2+1],lineList[i*2+2])
59  for i in range((len(lineList)-1)//2)])
60  for key in self.m_args:
61  if key in variables:
62  variables[key]=self.m_args[key]
63  self.m_vars=dict(list(self.m_vars.items())+list(variables.items()))
64  return True
65  else:
66  return False
