30  """Constructor
31  @param filename string file name of s-parameter file to read.
32  @param wfProjName string (optional, defaults to None) name of waveform reference designator in project
33  if the filename parameter is of a project that generates waveforms.
34  @param normalizedDCGain float (optional, defaults to None) DC gain to normalize response to.
35  @param multiplyByTs bool (optional, defaults to True) whether to multiply the waveform by the sample period
36  @param derivative bool (optional, defaults to False) whether to use the derivative of the waveform
37  @param **kwargs dict (optional, defaults to {}) dictionary of arguments for the file
38 
39  Reads the waveform file and produces SParameters.
40 
41  The s-parameters are for a two-port device that is "amplifier-like", meaning it has infinite
42  input impedance, zero output impedance, the impulse response forms s21, and has infinite reverse
43  isolation.
44 
45  If normalizedDCGain is 0 or None, the DC gain is not normalized, otherwise the filter is normalized by setting
46  the sum of the values in the impulse response equal to this gain value.
47 
48  if multiplyByTs is True, it assumes that the impulse response waveform is an ordinary waveform, and the impulse
49  response is formed by multiplying every value by the sample period.
50 
51  if derivative is True, the derivative of the waveform is taken. This is useful when you are provided a step
52  response.
53  """
54  ext=str.lower(filename).split('.')[-1]
55  if ext == 'si':
56  from SignalIntegrity.App.SignalIntegrityAppHeadless import ProjectWaveform
57  wf=ProjectWaveform(filename,wfProjName,None,**kwargs)
58  if wf == None:
59  raise SignalIntegrityExceptionWaveformFile('waveform could not be produced by '+filename)
60  else:
61  try:
62  wf=Waveform().ReadFromFile(filename)
63  except:
64  raise SignalIntegrityExceptionWaveformFile('waveform could not be produced by '+filename)
65  if derivative:
66  wf=wf.Derivative()
67  if multiplyByTs:
68  wf=wf*(1./wf.TimeDescriptor().Fs)
69  if not ((normalizedDCGain == None) or (normalizedDCGain == 0)):
