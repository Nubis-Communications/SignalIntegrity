220  def assigninvoke(self, tok, token = None):
221  "An assignment or a method invocation."
222  self.onelineblock(tok)
223  if not token:
224  token = tok.current()
225  token2 = tok.next()
226  if token2 == '=':
227  # assignment
228  return token + ' = ' + self.parser.parsevalue(tok)
229  if token2 == '.':
230  member = tok.next()
231  return self.assigninvoke(tok, token + '.' + member)
232  if token2 == '(':
233  parameters = self.parser.parseparameters(tok)
234  Trace.debug('Parameters: ' + parameters)
235  return self.assigninvoke(tok, token + parameters)
236  if token2 == '[':
237  square = self.parser.parseinsquare(tok)
238  return self.assigninvoke(tok, token + square)
239  if token2 == '{':
240  # ignore anonymous class
241  self.parser.parseupto('}', tok)
242  return token
243  if token2 == '++':
244  Trace.debug('Increasing invoked ' + token)
245  tok.autoincreases.append(token)
246  return self.assigninvoke(tok, token + ' + 1')
247  if token2 == '--':
248  Trace.debug('Decreasing invoked ' + token)
249  tok.autodecreases.append(token)
250  return self.assigninvoke(tok, token + ' - 1')
251  if token2 in [';', ',', ')']:
252  # finished invocation
253  return token
254  if token2 in tok.javasymbols:
255  Trace.error('Unknown symbol ' + token2 + ' for ' + token)
256  return '*error ' + token + ' ' + token2 + ' error*'
257  token3 = tok.next()
258  if token3 == ';':
259  # a declaration; ignore
260  return ''
261  if token3 == '=':
262  # declaration + assignment

265  if token3 == '[':
266  # array declaration
267  self.parser.parseupto(']', tok)
268  return self.assigninvoke(tok, token2)
