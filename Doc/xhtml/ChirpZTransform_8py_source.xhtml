25 def CZT(x,Fs,fs,fe,M,highSpeed=True):
26  """Provides M+1 points from fs to fe of the chirp z transform of the vector x
27  @param x an input vector (list) of real time-domain samples
28  @param Fs the sample rate of the the data samples in x
29  @param fs the start frequency for the result
30  @param fe the end frequency for the result
31  @param M the number of points in the result (-1)
32  @param highSpeed (optional, defaults to True) whether to use the high-speed,
33  but unreadable version. The readable, low-speed version is the definition
34  is used for testing.
35  @remark
36  Technically, the chirp-z transform can evaluate the z-transform at equi-angularly
37  spaced points on a somewhat arbitrary arc, but here, we use it to compute equi-angularly
38  spaced points on the rim of the unit circle, and as such, it is used simply to resample
39  data. Usually, we set fs=0 and the result is an M+1 point DFT of the waveform from 0 to
40  fs, without the same limitations imposed as the standard DFT.
41  The equation for the czt is:
42  \f$m\in 0\ldots M\f$
43  \f$X\left[m\right]=\sum_{k=0}^{K-1}x\left[k\right]\cdot e^{-j\cdot2\pi\cdot\frac{k\cdot m}{M}\cdot F_{e}}\f$
44  """
45  M=int(M); K=len(x); fs=float(fs); Fs=float(Fs); fe=float(fe)
46  theta0=fs/Fs; phi0=(fe-fs)/Fs*1./M; A0=1.; W0=1.
47  A=A0*cmath.exp(1j*2.*math.pi*theta0)
48  W=W0*cmath.exp(1j*2.*math.pi*phi0)
49  if highSpeed:
50  L=pow(2,int(math.ceil(math.log(float(M+K),2.))))*2
51  v=[0 for l in range(L)]
52  for k in range(M+1): v[k]=pow(W,float(k*k)/2.)
53  for k in range(M+1,L-K+1): v[k]=0.
54  for k in range(L-K+1,L): v[k]=pow(W,(L-k)*(L-k)/2.)
55  V=fft.fft(v)
56  y = [0 for l in range(L)]
57  for k in range(K): y[k]=x[k]*pow(A,-k)*pow(W,-(k*k)/2.)
58  for k in range(K,L): y[k]=0
59  Y=fft.fft(y)
60  P=[Y[l]*V[l] for l in range(L)]
61  P=fft.ifft(P)
62  X = [P[m]/v[m] for m in range(M+1)]
63  else:
64  Zeta=[A*pow(W,m) for m in range(M+1)]
65  X=[]
66  for m in range(M+1):
67  Xm=0.
68  for k in range(K): Xm=Xm+x[k]*pow(Zeta[m],-k)
69  X.append(Xm)
70  return X
