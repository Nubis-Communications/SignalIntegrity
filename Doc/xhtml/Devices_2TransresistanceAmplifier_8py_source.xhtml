43 def TransresistanceAmplifierFourPort(G,Zi,Zo,Z0=50.):
44  """TransresistanceAmplifierFourPort
45  Four port transresistance amplifier
46  @param G float transresistance gain
47  @param Zi float or complex input impedance
48  @param Zo float or complex output impedance
49  @param Z0 (optional) float reference impedance (defaults to 50 ohms)
50  @return list of list s-parameter matrix for a transresistance amplifier
51  @remark
52  The current sense element senses the current entering port 1 and exiting port 2
53  with a series resistance in the current sense path of Zi.\n
54  The voltage generated by the amplifier has the plus terminal at port 4 and
55  and the minus terminal at port 3 with Zo in series.\n
56  """
57  S11=Zi/(Zi+2.*Z0)
58  S12=2.*Z0/(Zi+2.*Z0)
59  S13=0
60  S14=0
61  S21=S12
62  S22=S11
63  S23=0
64  S24=0
65  S31=2.*G*Z0/((Zi+2.*Z0)*(Zo+2.*Z0))
66  S32=-S31
67  S33=Zo/(Zo+2.*Z0)
68  S34=2.*Z0/(Zo+2.*Z0)
69  S41=S32
70  S42=S31
71  S43=S34
72  S44=S33
73  return [[S11,S12,S13,S14],
74  [S21,S22,S23,S24],
75  [S31,S32,S33,S34],
76  [S41,S42,S43,S44]]

78 def TransresistanceAmplifierThreePort(G,Zi,Zo,Z0=50.):
79  """TransresistanceAmplifierThreePort
80  Three port transresistance amplifier
81  @param G float transresistance gain
82  @param Zi float or complex input impedance
83  @param Zo float or complex output impedance
84  @param Z0 (optional) float reference impedance (defaults to 50 ohms)
85  @return list of list s-parameter matrix for a transresistance amplifier
86  @remark
87  The three port transresistance amplifier is the same as the four port transresistance amplifier with
88  ports two and three connected together and exposed as a single port.\n
89  The current sense element senses the current entering port 3 and exiting port 1
90  with a series resistance in the current sense path of Zi.\n
91  The voltage generated by the amplifier has the plus terminal at port 2 and
92  and the minus terminal at port 3 with Zo in series.\n
93  """
94  D=3.*Z0*Z0+(2.*Zo+2.*Zi-G)*Z0+Zo*Zi
95  S11=(Zo*Zi+Z0*(2.*Zi-G)-Z0*Z0)/D
96  S12=(2.*Z0*Z0)/D
97  S13=(2.*Z0*Z0+2.*Zo*Z0)/D
98  S21=(2.*Z0*Z0+2.*G*Z0)/D
99  S22=(Zo*Zi+Z0*(2.*Zo-G)-Z0*Z0)/D
100  S23=(2.*Z0*Z0+Z0*(2.*Zi-2.*G))/D
101  S31=(2.*Z0*Z0+Z0*(2.*Zo-2.*G))/D
102  S32=(2.*Z0*Z0+2.*Zi*Z0)/D
103  S33=(Zo*Zi-Z0*Z0+G*Z0)/D
104  return [[S11,S12,S13],
105  [S21,S22,S23],
106  [S31,S32,S33]]

108 def TransresistanceAmplifierTwoPort(G,Zi,Zo,Z0=50.):
109  """TransresistanceAmplifierTwoPort
110  Two port transresistance amplifier
111  @param G float transresistance gain
112  @param Zi float or complex input impedance
113  @param Zo float or complex output impedance
114  @param Z0 (optional) float reference impedance (defaults to 50 ohms)
115  @return list of list s-parameter matrix for a transresistance amplifier
116  @remark
117  The two port transresistance amplifier is the same as the three port transresistance amplifier with
118  port 3 grounded.\n
119  The current sense element senses the current entering port 1
120  with a series resistance in the current sense path of Zi.\n
121  The voltage generated by the amplifier is at port 2 referenced to ground
122  with Zo in series with the output.\n
123  """
124  return [[(Zi-Z0)/(Zi+Z0),0.],[2.*G*Z0/((Zi+Z0)*(Zo+Z0)),(Zo-Z0)/(Zo+Z0)]]
