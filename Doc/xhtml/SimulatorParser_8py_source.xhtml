35  def _ProcessSimulatorLine(self,line):
36  """processes a line of a netlist, handing simulator specific commands
37  Lines that can be processed at this level are processed and lines that
38  are unknown are place in a list of unknown lines for upstream processing. This
39  enables derived classes to benefit from what this class knows how to process and
40  to simply add specific functionality. As a simple example, a derived simulator class
41  needs to add output probes, and this simple system description class knows nothing of
42  this.
43  netlist lines that are handled at this level are:
44  - 'output' - addition of an output probe
45  - 'voltageoutput' - addition of a voltage output probe with a name.
46  - 'voltagesource' or 'networkanalyzerport' - addition of a voltage source.
47  - 'currentsource' - addition of a current source.
48  Calls SystemDescriptionParser._ProcessLines(),
49  exludes 'connect' and 'port' in first call, then processes simulator lines, then
50  calls upstream one more time for the device connections, again excluding 'port'.
51  """
52  lineList=self.ReplaceArgs(line.split())
53  if len(lineList) == 0: return
54  elif lineList[0] == 'output':
55  if self.m_sd.pOutputList is None: self.m_sd.pOutputList = []
56  for i in range(1,len(lineList),2):
57  self.m_sd.pOutputList.append((lineList[i],int(lineList[i+1])))
58  # pragma: silent exclude
59  elif lineList[0] == 'voltageoutput':
60  if self.m_sd.pOutputList is None: self.m_sd.pOutputList = []
61  for i in range(1,len(lineList),3):
62  self.m_sd.pOutputList.append((lineList[i+1],int(lineList[i+2]),lineList[i]))
63  # pragma: include
64  elif lineList[0] in ['voltagesource','networkanalyzerport']:
65  self.m_sd.AddVoltageSource(lineList[1],int(lineList[2]))
66  elif lineList[0] == 'currentsource':
67  self.m_sd.AddCurrentSource(lineList[1],int(lineList[2]))
68  else: self.m_ul.append(line)
69  def _ProcessLines(self):
70  """processes all of the lines in a netlist
71  @see _ProcessLine() for explanation of parameters and functionality.
72  """
73  SystemDescriptionParser._ProcessLines(self,['connect','port'])
74  self.m_sd = Simulator(self.m_sd)
75  lines=copy.deepcopy(self.m_ul); self.m_ul=[]
76  for line in lines: self._ProcessSimulatorLine(line)
77  lines=copy.deepcopy(self.m_ul); self.m_ul=[]
78  for line in lines: SystemDescriptionParser._ProcessLine(self,line,['port'])
79  return self
