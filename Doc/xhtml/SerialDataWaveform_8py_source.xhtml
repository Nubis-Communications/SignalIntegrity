27 class SerialDataWaveform(Waveform):
28  """a serial data NRZ waveform"""
29  rtvsT=0.5903445
30  def __init__(self,pattern,bitRate,amplitude=1.0,risetime=0.,delay=0.,tdOrFs=None):
31  """constructor
32  @param pattern list of bits in pattern which must be 0 or 1
33  @param bitRate float bitrate in bits/s
34  @param amplitude (optional, defaults to 1.0) float amplitude of zero centered serial data waveform (pk-pk amplitude is twice this value)
35  @param risetime (optional, defaults to 0.0) float risetime of a 1/2 cosine response.
36  @param delay (optional, defaults to 0,0) float delay in time into the pattern.
37  @param tdOrFs (optional, defaults to None) time descriptor or float sample rate. If None is specified, then
38  the descriptor is generated for one pattern length with a sample rate of 10x the bitrate. Otherwise, if a float
39  is specified, then a time descriptor for one pattern length is generated with the sample rate specified. Otherwise
40  the time descriptor specified is used.
41  @return self, a waveform.
42  @throw SignalIntegrityExceptionWaveform exception is raised if the risetime exceeds 59% of the unit interval as this causes
43  the calculation to fail.
44  @todo the failure for a given risetime is due to the simplicity of the algorithm which should be improved at some point.
45  this simplicity is that it only looks at adjacent bits to determine the intersymbol effect of the risetime.
46  @note the left edge of the first bit in the pattern is at the delay time. But because a cosine is used for
47  the risetime emulation, the 50% point of the cosine is reached at this edge.
48  @note the risetime is adjusted if it is too low relative to the sample period.
49  """
50  if tdOrFs is None:
51  sampleRate=10.*bitRate
52  td=self.TimeDescriptor(bitRate, sampleRate, len(pattern))
53  elif isinstance(tdOrFs,float):
54  sampleRate=tdOrFs
55  td=self.TimeDescriptor(bitRate, sampleRate, len(pattern))
56  else:
57  td=tdOrFs
58  T=max(risetime/self.rtvsT,1/td.Fs)
59  patternTimeLength=len(pattern)/bitRate
60  unitInterval=1./bitRate
61  if T>unitInterval:
62  # todo: would like to handle this case in the future
63  raise SignalIntegrityExceptionWaveform('PRBS risetime too high\nfor waveform generation. '+\
64  'Limit is '+ToSI(unitInterval*self.rtvsT,'s')+' for given bitrate')
65  v=[0. for _ in range(len(td))]
66  for k in range(len(td)):
67  t=td[k]
68  timeInPattern=((t-delay)%patternTimeLength+patternTimeLength)%patternTimeLength
69  thisBit=int(math.floor(timeInPattern/unitInterval))
70  timeInBit=timeInPattern-thisBit*unitInterval
71  thisBitValue=pattern[thisBit]
72  if timeInBit>=T/2. and (timeInBit-unitInterval)<=-T/2.:
73  v[k]=2.*amplitude*(thisBitValue-0.5)
74  elif timeInBit<T/2.:
75  previousBit=(thisBit-1+len(pattern))%len(pattern)
76  previousBitTransition=thisBitValue-pattern[previousBit]
77  if previousBitTransition==0:
78  v[k]=2.*amplitude*(thisBitValue-0.5)
79  else:
80  v[k]=amplitude*previousBitTransition*math.cos(math.pi*(timeInBit/T+3./2.))
81  elif (timeInBit-unitInterval)>-T/2.:
82  nextBit=(thisBit+1)%len(pattern)
83  nextBitTransition=pattern[nextBit]-thisBitValue
84  if nextBitTransition==0:
85  v[k]=2.*amplitude*(thisBitValue-0.5)
86  else:
87  v[k]=amplitude*nextBitTransition*math.cos(math.pi*((timeInBit-unitInterval)/T+3./2.))
88  Waveform.__init__(self,Waveform(td,v))
