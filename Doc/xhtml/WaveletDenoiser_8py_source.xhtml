35  def DenoisedWaveform(wf,pct=30.,mult=5.,isDerivative=True):
36  """Denoises a waveform.
37  @details The noise is estimated by the amount of noise in the last pct percent of the
38  last scale of the wavelet transform. The threshold is set as a multiplier on
39  the noise, generally 5 is used. Thus wavelets more than 5 times the noise floor
40  are kept, while all others are deleted (this is called hard-thresholding).\n
41  Usually this algorithm is used to denoise TDR waveforms, which are usually steps
42  and the denoising is performed on the derivative of the step waveform. Therefore,
43  the noise threshold needs to be shaped based on the effect of the derivative on
44  the noise shape. For impulsive waveforms, no derivative is performed and
45  isDerivative is set to False, causing a white noise threshold to be assumed.
46  @param wf instance of class Waveform to be denoised.
47  @param pct (optional) float percentage (defaults to 30).
48  @param mult (optional) float noise multiplier for threshold (defaults to 5).
49  @param isDerivative (optional) bool whether the waveform is a derivative
50  (defaults to True).
51  @return the denoised waveform.
52  @remark the algorithm assumes there is no signal in the last pct percent
53  of the last scale of the dwt
54  @see Wavelet
55  """
56  w=WaveletDenoiser.wavelet
57  Ki=wf.td.K; Kf=int(pow(2,math.ceil(log2(wf.td.K))))
58  PadLeft=Kf-Ki; pct=pct*Ki/Kf; pwf=wf*WaveformTrimmer(-PadLeft,0)
59  X=w.DWT(pwf.Values())
60  T=WaveletDenoiser.DerivativeThresholdCalc(X,w.h,pct,isDerivative)

79  def DerivativeThresholdCalc(X,h,pct=30.,isDerivative=True):
80  L=len(h)
81  K=len(X)
82  N=K//2
83  B=int(log2(K)-log2(L))+1
84  if isDerivative:
85  E=[math.sqrt(2.*(1-math.cos(math.pi*n/N))) for n in range(N+1)]
86  TS=WaveletDenoiser.DWTNoiseShapeCalc(E,h)
87  else:
88  TS=[1 for n in range(N+1)]
89  P=int(math.floor(pct/100.*K/2.))
90  sigmaw=std(X[K-P:K])
91  sigma=sigmaw*math.sqrt(N)/TS[B-1]
92  th=[sigma/math.sqrt(N) for k in range(K)]
93  for k in range(K):
94  # warning - seems to depend on L a power of 2
95  b=int(math.floor(log2(max(L//2,k)))-log2(L//2))
96  th[k]=th[k]*TS[b]
97  return th
98  @staticmethod
99  def DWTNoiseShapeCalc(E,h):
100  L=len(h)
101  N=len(E)-1
102  B=int(log2(N)-log2(L)+2)
103  S=[0 for _ in range(B)]
104  for i in range(B-1):
105  accn=0.
106  for n in range(N+1):
107  acc=0.
108  for l in range(L):
109  acc=acc+pow(-1,l)*h[L-1-l]*cmath.exp(-1j*math.pi*n*l/N)
110  accn=accn+pow(abs(E[n])*abs(acc),2)
111  S[B-1-i]=math.sqrt(accn)
112  for n in range(N//2+1):
113  accl=0
114  accr=0
115  for l in range(L):
116  accl=accl+h[l]*cmath.exp(-1j*math.pi*n*l/N)
117  accr=accr+h[l]*cmath.exp(-1j*math.pi*(N-n)*l/N)
118  E[n]=math.sqrt(pow(abs(E[n])*abs(accl),2.)+pow(abs(E[N-n])*abs(accr),2.))
119  N=N//2
120  acc=0
121  for n in range(N+1):
122  acc=acc+pow(abs(E[n]),2)
123  S[0]=math.sqrt(acc)
124  return S
