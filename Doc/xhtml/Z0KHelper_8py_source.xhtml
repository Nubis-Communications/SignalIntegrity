21 from numpy import diag

24 def Z0KHelper(Z0K,P):
25  """Resolves reference impedance and scaling factor from a specification
26  @param Z0K tuple containing (Z0,K) - the reference impedance and scaling factor
27  @param P integer number of ports
28  @remark
29  This very useful function helps resolve all of the possibilities for
30  specification of reference impedance and scaling factor under the multitude
31  of possibilities on how it can be provided to various conversion functions
32 
33  It operates by selecting the best possible choices with the specification
34  of the least information.
35 
36  If Z0 is not specified, the default 50 Ohms is selected and if the
37  scaling factor is not specified, the default sqrt(Z0) is chosen.
38  These are provided in matrix form when needed and complex when needed
39 
40  both Z0 and K may be specified as lists, in which case they represent
41  port reference impedance and scaling factors
42  """
43  (Z0,K)=Z0K
44  if Z0 is None:
45  Z0=matrix(diag([50.0]*P))
46  elif isinstance(Z0,list):
47  Z0=matrix(diag([float(i.real)+float(i.imag)*1j for i in Z0]))
48  elif isinstance(Z0.real,float) or isinstance(Z0.real,int):
49  Z0=matrix(diag([float(Z0.real)+float(Z0.imag)*1j]*P))
50  if K is None:
51  K=sqrt(Z0)
52  elif isinstance(K,list):
53  K=matrix(diag([float(i.real)+float(i.imag)*1j for i in K]))
54  elif isinstance(K.real,float) or isinstance(K.real,int):
55  K=matrix(diag([float(K.real)+float(K.imag)*1j]*P))
56  return (Z0,K)
