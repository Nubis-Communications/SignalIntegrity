36 def Mutual(Ll,Lr,M,f,Z0=None,K=None):
37  """Mutual
38  Mutual Inductance
39  @param Ll float self inductance of left leg
40  @param Lr float self inductance of right leg
41  @param M float mutual inductance between legs
42  @param f float frequency
43  @param Z0 (optional) float or complex reference impedance (assumed 50 ohms)
44  @param K (optional) float or complex scaling factor (actually unused).
45  @return list of list representing s-parameter matrix of a mutual inductance
46  @remark The device is four port.\n
47  The left leg is from port 1 to 2.\n
48  The right leg is from port 3 to 4.\n
49  The arrow for the mutual points to ports 1 and 3.\n
50  @todo use Z0KHelper to resolve reference impedance and scaling factor. Currently K is not used.
51  @todo remove old mutual inductance code
52  """
53  s=1j*2.*math.pi*f
54  D=s*s*(Ll*Lr-M*M)+2*Z0*s*(Ll+Lr)+4*Z0*Z0
55  S11=(s*s*(Ll*Lr-M*M)+2*s*Ll*Z0)/D
56  S12=(2*Z0*(s*Lr+2*Z0))/D
57  S13=(2*s*M*Z0)/D
58  S14=-S13
59  S21=S12
60  S22=S11
61  S23=S14
62  S24=S13
63  S31=S13
64  S32=S23
65  S33=(s*s*(Ll*Lr-M*M)+2*s*Lr*Z0)/D
66  S34=2*Z0*(s*Ll+2*Z0)/D
67  S41=S14
68  S42=S24
69  S43=S34
70  S44=S33
71  return [[S11,S12,S13,S14],
72  [S21,S22,S23,S24],
73  [S31,S32,S33,S34],
74  [S41,S42,S43,S44]]
