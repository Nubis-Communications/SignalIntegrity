37 def Mutual(Ll,Lr,M,f,Z0=None,K=None):
38  """Mutual
39  Mutual Inductance
40  @param Ll float self inductance of left leg
41  @param Lr float self inductance of right leg
42  @param M float mutual inductance between legs
43  @param f float frequency
44  @param Z0 (optional) float or complex reference impedance (assumed 50 Ohms)
45  @param K (optional) float or complex scaling factor (actually unused).
46  @return list of list representing s-parameter matrix of a mutual inductance
47  @remark The device is four port.\n
48  The left leg is from port 1 to 2.\n
49  The right leg is from port 3 to 4.\n
50  The arrow for the mutual points to ports 1 and 3.\n
51  @todo use Z0KHelper to resolve reference impedance and scaling factor. Currently K is not used.
52  @todo remove old mutual inductance code
53  """
54  s=1j*2.*math.pi*f
55  D=s*s*(Ll*Lr-M*M)+2*Z0*s*(Ll+Lr)+4*Z0*Z0
56  S11=(s*s*(Ll*Lr-M*M)+2*s*Ll*Z0)/D
57  S12=(2*Z0*(s*Lr+2*Z0))/D
58  S13=(2*s*M*Z0)/D
59  S14=-S13
60  S21=S12
61  S22=S11
62  S23=S14
63  S24=S13
64  S31=S13
65  S32=S23
66  S33=(s*s*(Ll*Lr-M*M)+2*s*Lr*Z0)/D
67  S34=2*Z0*(s*Ll+2*Z0)/D
68  S41=S14
69  S42=S24
70  S43=S34
71  S44=S33
72  return [[S11,S12,S13,S14],
73  [S21,S22,S23,S24],
74  [S31,S32,S33,S34],
75  [S41,S42,S43,S44]]
