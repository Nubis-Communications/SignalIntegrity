21 class CalibrationMeasurement(object):
22  """Base class for calibration measurements"""
23  def __init__(self,type,name=None):
24  """Constructor
25  @param type string representing the type of the measurement.
26  @param name (optional) string representing the name of the measurement
27  @remark
28  The name of the measurement is not used for anything, but can be used to identify
29  the name of a calibration measurement externally.
30 
31  valid types of calibration measurements are:
32 
33  - 'reflect' - a reflect measurement taken on a port with a reflect standard
34  (like short, open, or load).
35  - 'thru' - a thru calibration measurement taken between two ports.
36  - 'xtalk' - a crosstalk calibration measurement typically taken between two ports that
37  are completely unconnected.
38  @see ReflectCalibrationMeasurement
39  @see ThruCalibrationMeasurement
40  @see XtalkCalibrationMeasurement
41  """
42  self.type=type
43  self.name=name
