78  def extractcommand(self, pos):
79  "Extract the command from elyxer.the current position."
80  if not pos.checkskip(FormulaCommand.start):
81  pos.error('Missing command start ' + FormulaCommand.start)
82  return
83  if pos.finished():
84  return self.emptycommand(pos)
85  if pos.current().isalpha():
86  # alpha command
87  command = FormulaCommand.start + pos.globalpha()

89  pos.checkskip('*')

91  # symbol command
