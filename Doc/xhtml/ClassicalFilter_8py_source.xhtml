24 class ClassicalFilter(SParameters):
25  """Classical filter"""
26  def __init__(self,f,z,p,k,f0=1.,Z0=50.):
27  self.z=z
28  self.p=p
29  self.k=k
30  self.f0=f0
31  SParameters.__init__(self,f,None,Z0)
32  def _Evaluate(self,f):
33  """Evaluate filter
34  @param f float frequency
35  @return filter evaluated at f
36  """
37  s=1j*f/self.f0
38  D=np.prod([s-self.p[k] for k in range(len(self.p))]); N=np.prod([s-self.z[k] for k in range(len(self.z))])
39  return self.k*N/D
40  def __getitem__(self,n):
41  return [[1.,0],[2.*self._Evaluate(self.m_f[n]),-1.]]
