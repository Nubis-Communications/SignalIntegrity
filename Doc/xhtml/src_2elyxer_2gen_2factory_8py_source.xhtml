115  "A parsing tree"
116 
117  default = '~~default~~'

119  def __init__(self, types):
120  "Create the parse tree"
121  self.root = dict()

125  def addstart(self, type, start):
126  "Add a start piece to the tree"
127  tree = self.root
128  for piece in start.split():
129  if not piece in tree:

131  tree = tree[piece]
132  if ParseTree.default in tree:

134  tree[ParseTree.default] = type

137  "Find the current sentence in the tree"
138  branches = self.matchline(reader.currentline())
139  while not ParseTree.default in branches[-1]:
