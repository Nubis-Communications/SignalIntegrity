29 class ImpedanceProfile(list):
30  """ Impedance profile generated using the peeling algorithm. After construction, it is
31  a list of rho values"""
32  rhoLimit=0.99
33  ZLimit=10e3
34  def __init__(self,sp,sections,port):
35  """ Constructor\n
36  Generates a list of rho values for a specified number of sections by peeling the
37  s-parameters provided from the port provided.
38  @param sp instance of class SParameters
39  @param sections integer number of sections to compute
40  @param port port of the SParameters instance to peel from
41  """
42  list.__init__(self,[])
43  N = len(sp)-1
44  self.m_Td = 1./(4.*sp.f()[N])
45  self.m_Z0 = sp.m_Z0
46  fr=sp.FrequencyResponse(port,port)
47  self.m_fracD=fr._FractionalDelayTime()
48  fr=fr._DelayBy(-self.m_fracD)
49  S11 = fr.Values()
50  zn2 = [cmath.exp(-1j*2.*math.pi*n/N*1/2) for n in range(N+1)]
51  finished=False
52  rho=0.0
53  for _ in range(sections):
54  if finished:
55  self.append(rho)
56  continue
57  rho = 1/(2.*N)*(S11[0].real + S11[N].real +
58  sum([2.*S11[n].real for n in range(1,N)]))
59  rho=max(-self.rhoLimit,min(rho,self.rhoLimit))
60  self.append(rho)
61  if abs(rho)==self.rhoLimit:
62  finished=True
63  continue
64  rho2=rho*rho
65  S11=[(-S11[n]+S11[n]*rho2*zn2[n]-rho*zn2[n]+rho)/
66  (rho2+S11[n]*rho*zn2[n]-S11[n]*rho-zn2[n])
67  for n in range(N+1)]
68  def Z(self):
69  """impedance list
70  @return list of impedances, in ohms, for each section.
71  """
72  return [max(0.,min(self.m_Z0*(1+rho)/(1-rho),self.ZLimit))
73  for rho in self]
74  def DelaySection(self):
75  """time length of each section
76  @return float time length of each section.
77  """
78  return self.m_Td
79  def SParameters(self,f):
80  """s-parameters
81  @param f list of frequencies for the s-parameters
82  @return instance of class SParameters calculated by aggregating the list of rho
83  values using lossless transmission line sections with the rho calculated for each
84  section.
85  """
86  N = len(f)-1
87  Gsp=[None for n in range(N+1)]
88  gamma=[1j*2.*math.pi*f[n]*self.m_Td for n in range(N+1)]
89  for n in range(N+1):
90  tacc=matrix([[1.,0.],[0.,1.]])
91  for m in range(len(self)):
92  tacc=tacc*matrix(S2T(IdealTransmissionLine(self[m],gamma[n])))
93  Gsp[n]=T2S(tacc.tolist())
94  sp = SParameters(f,Gsp,self.m_Z0)
95  return sp
