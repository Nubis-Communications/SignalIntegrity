35  def PropertyString(self,stype='raw'):
36  if stype=='attr':
37  result=''
38  if self.GetValue('Visible') and not self.GetValue('Hidden'):
39  if self.GetValue('KeywordVisible'):
40  if self.GetValue('Keyword') != None and self.GetValue('Keyword') != 'None':
41  result=result+self.GetValue('Keyword')+' '
42  if self.GetValue('Type')=='string':
43  value = self.GetValue('Value')
44  elif self.GetValue('Type')=='file':
45  value=('/'.join(str(self.GetValue('Value')).split('\\'))).split('/')[-1]
46  elif self.GetValue('Type')=='int':
47  value = self.GetValue('Value')
48  elif self.GetValue('Type')=='float':
49  value = str(ToSI(float(self.GetValue('Value')),self.GetValue('Unit')))
50  else:
51  value = str(self.GetValue('Value'))
52  result=result+value
53  return result
54  elif stype == 'raw':
55  if self.GetValue('Type')=='string':
56  value = self.GetValue('Value')
57  elif self.GetValue('Type')=='file':
58  value = self.GetValue('Value')
59  elif self.GetValue('Type')=='int':
60  value = self.GetValue('Value')
61  elif self.GetValue('Type')=='float':
62  unitstr=self.GetValue('Unit')
63  # check for scaling of units (indicated by an exponent on the front of the units)
64  factor=''
65  if len(unitstr)>=2:
66  if unitstr[0:2]=='e-':
67  # this number is scaled - extract the scaling factor
68  factor=unitstr.split(' ')[0] # this will be something like 'e-34'
69  value = str(float(self.GetValue('Value')+factor))
70  else:
71  value = str(self.GetValue('Value'))
72  return value
73  elif stype == 'entry':
74  if self.GetValue('Type')=='string':
75  value = str(self.GetValue('Value'))
76  elif self.GetValue('Type')=='file':
77  value = str(self.GetValue('Value'))
78  elif self.GetValue('Type')=='int':
79  value = str(self.GetValue('Value'))
80  elif self.GetValue('Type')=='float':
81  value = str(ToSI(float(self.GetValue('Value')),self.GetValue('Unit')))
82  else:
83  value = str(self.GetValue('Value'))
84  return value
85  elif stype == 'netlist':
86  if self.GetValue('Type')=='string':
87  value = self.GetValue('Value')
88  elif self.GetValue('Type')=='file':
89  value=self.GetValue('Value')
90  if not value is None:
91  if ' ' in value:
92  value = "'"+value+"'"
93  elif self.GetValue('Type')=='int':
94  value = self.GetValue('Value')
95  elif self.GetValue('Type')=='float':
96  value = str(ToSI(float(self.GetValue('Value')),self.GetValue('Unit'),letterPrefixes=False))
97  else:
98  value = str(self.GetValue('Value'))
99  return value
100  else:
101  raise ValueError
102  return str(self['Value'])
103  def SetValueFromString(self,string):
104  if self['Type']=='string' or self['Type']=='enum':
105  self['Value']=str(string)
106  elif self.GetValue('Type')=='file':
107  self['Value']=str(string)
