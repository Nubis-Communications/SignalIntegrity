26  """process transfer matrices

42  def ProcessWaveforms(self,wfl,td=None,adaptToLargest=False):
43  """processes input waveforms and produces output waveforms
44  @param wfl list of Waveform input waveforms to process. If numbers are in the
45  waveform list, they are assumed to be DC waveforms, that are treated specially.
46  @param td (optional) instance of class TimeDescriptor. If this is included,
47  all final waveforms are adapted to this descriptor. Otherwise, the sample rates of
48  each waveform are used in the computation of the impulse responses.
49  @param adaptToLargest bool (optional, defaults to False) causes waveform adaption
50  during summation to choose the waveform with the largest absolute value of a point.
51  This helps when models have frequency responses near the end frequency and causes
52  the adaption to resample the smaller waveforms, which have less effect.
53  @remark Externally, the order of the input and output waveforms are known. The
54  input waveforms must be provided in that order and the output waveforms are
55  produced in that order.
56  """
57  if td is None:
58  td = [wflm.td.Fs if isinstance(wflm,Waveform) else None for wflm in wfl]
59  ir = self.TransferMatrices.ImpulseResponses(td)
60  fr = self.TransferMatrices.FrequencyResponses() # for DC inputs
61  result=[]
62  for o in range(len(ir)):
63  acc=[]
64  for i in range(len(ir[o])):
65  acc.append(ir[o][i].FirFilter().FilterWaveform(wfl[i])
66  if isinstance(wfl[i],Waveform)
67  else (wfl[i]*fr[o][i][0]).real)
68  # pragma: silent exclude
69  if self.HasACallBack():
70  progress=(float(o)/len(ir)+float(i)/(len(ir)*len(ir[o])))*100.0
71  if not self.CallBack(progress):
72  raise SignalIntegrityExceptionSimulator('calculation aborted')
73  if adaptToLargest and len(acc)>1:
74  largestValue=0.0; largestIndex=0
75  for wfi in range(len(acc)):
76  if isinstance(acc[wfi],Waveform):
77  absMax=max(acc[wfi].Values('abs'))
78  if absMax>=largestValue:
79  largestIndex=wfi; largestValue=absMax
80  acc=[acc[largestIndex]]+acc[0:largestIndex]+acc[largestIndex+1:]
81  # if the first element of the accumulator is not a waveform (a number, assuming a DC value),
82  # then rearrange so a waveform comes first.
83  if not isinstance(acc[0],Waveform):
84  for wfi in range(len(acc)):
85  if isinstance(acc[wfi],Waveform):
86  acc=[acc[wfi]]+acc[0:wfi]+acc[wfi+1:]
87  break
88  # pragma: include
89  result.append(sum(acc))
90  return result
