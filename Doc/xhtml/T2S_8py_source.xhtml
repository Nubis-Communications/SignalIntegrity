24 def T2S(T,lp=None,rp=None):
25  """Converts generalized T-parameters to s-parameters
26  @param T list of list representing T-parameter matrix to convert
27  @param lp (optional) a list of left port numbers
28  @param rp (optional) a list of right port numbers
29  @note if no list of left and right ports are specified, it assumes that the first
30  P/2-1 port are on the left and the remaining ports are on the right
31  @remark Supports multi-port devices
32  @attention The number of ports must be even
33  @attention The port number in the lists are one-based (not zero-based)
34  @note The reference impedance and scaling factor associated with the s-parameters
35  is unchanged."""
36  P=len(T)//2+len(T[0])//2
37  if not isinstance(lp,list):
38  lp=range(1,P//2+1)
39  rp=range(P//2+1,P+1)
40  I=identity(P).tolist()
41  SL=[]
42  SR=[]
43  for p in range(P):
44  if (p+1) in rp:
45  SL.append(I[2*rp.index(p+1)+1])
46  SR.append(I[2*rp.index(p+1)])
47  else:
48  SL.append(T[2*lp.index(p+1)])
49  SR.append(T[2*lp.index(p+1)+1])
50  return (array(SL).dot(inv(array(SR)))).tolist()
