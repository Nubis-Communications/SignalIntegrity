25 def T2S(T,lp=None,rp=None):
26  """Converts generalized T-parameters to s-parameters
27  @param T list of list representing T-parameter matrix to convert
28  @param lp (optional) a list of left port numbers
29  @param rp (optional) a list of right port numbers
30  @note if no list of left and right ports are specified, it assumes that the first
31  P/2-1 port are on the left and the remaining ports are on the right
32  @remark Supports multi-port devices
33  @attention The number of ports must be even
34  @attention The port number in the lists are one-based (not zero-based)
35  @note The reference impedance and scaling factor associated with the s-parameters
36  is unchanged."""
37  P=len(T)//2+len(T[0])//2
38  if not isinstance(lp,list):
39  lp=range(1,P//2+1)
40  rp=range(P//2+1,P+1)
41  I=identity(P).tolist()
42  SL=[]
43  SR=[]
44  for p in range(P):
45  if (p+1) in rp:
46  SL.append(I[2*rp.index(p+1)+1])
47  SR.append(I[2*rp.index(p+1)])
48  else:
49  SL.append(T[2*lp.index(p+1)])
50  SR.append(T[2*lp.index(p+1)+1])
51  return array(matrix(SL)*matrix(SR).getI()).tolist()
