96  def Pattern(self,bits=None):
97  """generate a list of pseudo-random bits according to the polynomial
98  @param bits (optional, defaults to None) integer number of bits to return.
99  If bits are specified, then the pattern returned will contain the number of bits specified.
100  @return returns a list of integer (1 or 0) corresponding to the pattern.
101 
102  for an order P polynomial, the list of bits is 2^P-1 elements long, assuming bits are specified as None.
103  """
104  order = self.Order()
105  length=self.PatternLength() if bits == None else min(self.PatternLength(),bits)
106  pattern = [1 if k < order else 0 for k in range(length)]
107  for i in range(order,length):
108  pattern[i]=sum([self[order-p]*pattern[i-(order-p)] for p in range(order)])%2
109  pattern = [(b+1)%2 for b in pattern]
110  return pattern
