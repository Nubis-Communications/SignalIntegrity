24 class ResultsCache(object):
25  """base class for results caching
26  @note derived class must implement the function HashValue(), which determines the hash
27  corresponding to a definition.
28  """
29  def __init__(self,name,filename=None):
30  """constructor\n
31  When a project with a given filename is processed, various results in that project can be cached.
32  @param name string name of thing to cache. Examples are 'SParameters' and 'TransferMatrices'.
33  @param filename string base filename of project being processed.
34  """
35  self.filename=filename
36  self.extra='_cached'+name+'.p'
37  def HashValue(self,stuffToHash=''):
38  """ Generates the hash for a definition\n
39  @param stuffToHash repr of stuff to hash
40  @remark derived classes should override this method and call the base class HashValue with their stuff added
41  @return integer hash value
42  """
43  return hashlib.sha256(stuffToHash.encode()).hexdigest()
44  def CheckCache(self):
45  """Called to see if the cache has results that can be used instead of processing the result.\n
46  It calculates a hash value for the definition of the processing and sees if a _pickle_ containing
47  a cached result exists and can be loaded. Then it checks the times of the cache file and the
48  various subcomponents. Finally, if the hash value matches the cache, meaning they were both
49  generated from the same definition, it returns True with it's internal dictionary of the cached
50  results initialized. Otherwise, it returns False.
51  @return bool whether the cache can be used.
52  """
53  self.hash=self.HashValue()
54  import os
55  if self.filename is None:
56  return False
57  if not os.path.exists(self.filename+self.extra):
58  return False
59  if not self.CheckTimes(self.filename+self.extra):
60  return False
61  try:
62  with open(self.filename+self.extra,'rb') as f: tmp_dict = pickle.load(f)
63  except:
64  return False
65  try:
66  if tmp_dict['hash'] == self.hash:
67  self.__dict__.update(tmp_dict)
68  return True
69  return False
70  except:
71  return False

74  @return self
75  @note that the hash value for the result was generated through a previous call to CheckCache().
76  In other words, each cached value must be stored with a hash corresponding to the definition that generated
77  the result to be cached (to be checked when an attempt is made to load the cache). This hash is generated
78  automatically when the CheckCache call is made.
79  @see CheckCache()
80  """
81  if self.filename is None:
82  return
83 
84  members = [attr for attr in dir(self) if not callable(getattr(self, attr)) and not attr.startswith("__")]
85  pickleDict = {x:self.__dict__[x] for x in members}
86  with open(self.filename+self.extra, 'wb') as f: pickle.dump(pickleDict, f, 2)
87  return self
