27 class Gaussian(FirFilter):
28  """
29  Gaussian COM based FIR filter that, when convolved with step, provides
30  the risetime specified at the given sample rate.
31  """
32  a2080=1.683244 # hardcoded relationship that produces 20-80 risetime
33  a1090=2.563103 # hardcoded relationship that produces 10-90 risetime
34  def __init__(self,rt,Fs,is1090=True):
35  """constructor
36  @param rt float risetime desired.
37  @param Fs float sample rate
38  @param is1090 boolean (optional, defaults to True) whether 10-90 risetime
39  is specified. Otherwise 20-80 risetime is used.
40  """
41  a=self.a1090 if is1090 else self.a2080
42  impulse_response_length = rt*1.2*10*2
43  if impulse_response_length == 0:
44  FirFilter.__init__(self,FilterDescriptor(1,0,0),[1])
45  else:
46  import numpy as np
47  frequency_resolution = 1/impulse_response_length
48  frequency_points = np.ceil(Fs/2/frequency_resolution)-1
49  fd=EvenlySpacedFrequencyList(Fs/2,frequency_points)
50  v=[np.exp(-2*(np.pi*f*rt/a)**2) for f in fd]
51  fr=FrequencyResponse(fd,v)
52  ir=fr.ImpulseResponse()
53  fir=ir.FirFilter().NormalizeUnityDCGain()
54  FirFilter.__init__(self,fir.FilterDescriptor(),fir.FilterTaps())
