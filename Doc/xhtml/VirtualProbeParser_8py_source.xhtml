39  """processes a line of a netlist, handing virtual probe specific commands
40 
41  Lines that can be processed at this level are processed and lines that
42  are unknown are place in a list of unknown lines for upstream processing. This
43  enables derived classes to benefit from what this class knows how to process and
44  to simply add specific functionality. As a simple example, a derived simulator class
45  needs to add output probes, and this simple system description class knows nothing of
46  this.
47 
48  netlist lines that are handled at this level are:
49  - 'output' - addition of an output probe
50  - 'meas' - addition of a measurement probe.
51  - 'stim' - addition of a stimulus source.
52  - 'stimdef' - definition of a stimdef (the relationship between stims).
53 
54  """
55  lineList=self.ReplaceArgs(line.split())
56  if len(lineList) == 0: # pragma: no cover
57  return
58  if lineList[0] == 'meas':
59  if self.m_sd.pMeasurementList is None:
60  self.m_sd.pMeasurementList = []
61  for i in range(1,len(lineList),2):
62  self.m_sd.pMeasurementList.append((lineList[i],int(lineList[i+1])))
63  elif lineList[0] == 'output':
64  if self.m_sd.pOutputList is None:
65  self.m_sd.pOutputList = []
66  for i in range(1,len(lineList),2):
67  self.m_sd.pOutputList.append((lineList[i],int(lineList[i+1])))
68  elif lineList[0] == 'stim':
69  for i in range((len(lineList)-1)/3):
70  self.m_sd.AssignM(lineList[i*3+2],int(lineList[i*3+3]),lineList[i*3+1])
71  elif lineList[0] == 'stimdef':
72  self.m_sd.pStimDef = [[float(e) for e in r] for r in [s.split(',')
73  for s in ''.join(lineList[1:]).strip(' ').strip('[[').
74  strip(']]').split('],[') ]]
75  else:
76  self.m_ul.append(line)
