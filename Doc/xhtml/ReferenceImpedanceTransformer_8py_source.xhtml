23 def ReferenceImpedanceTransformer(Z0f,Z0i=None,Kf=None,Ki=None):
24  """ReferenceImpedanceTransformer
25  Reference impedance transformer
26  @param Z0f real or complex output reference impedance.
27  @param Z0i real or complex input reference impedance.
28  @param Kf (optional) real or complex scaling factor for the output (defaults to sqrt(Z0f))
29  @param Ki (optional) real or complex scaling factor for the input (defaults to sqrt(Z0i))
30  @return the list of list s-parameter matrix for a reference impedance transformer.
31  @todo put Z0i=50.0 in the input arguments and remove check within the code
32  @todo needs port numbering
33  """
34  Z0f=float(Z0f.real)+float(Z0f.imag)*1j
35  if Z0i is None:
36  Z0i=50.0
37  Z0i=float(Z0i.real)+float(Z0i.imag)*1j
38  if Kf is None:
39  Kf=cmath.sqrt(Z0f)
40  if Ki is None:
41  Ki=cmath.sqrt(Z0i)
42  Kf=float(Kf.real)+float(Kf.imag)*1j
43  Ki=float(Ki.real)+float(Ki.imag)*1j
44  p=(Z0f-Z0i)/(Z0f+Z0i)
45  return [[p,(1.0-p)*Kf/Ki],[(1.0+p)*Ki/Kf,-p]]
