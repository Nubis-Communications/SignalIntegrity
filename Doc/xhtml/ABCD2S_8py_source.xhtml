23 def ABCD2S(ABCD,Z0=None,K=None):
24  """Converts ABCD parameters to s-parameters.\n
25  @param ABCD list of list representing ABCD matrix to convert.
26  @param Z0 (optional) float or complex or list of reference impedance (defaults to None).
27  @param K (optional) float or complex or list of scaling factor (defaults to None).
28  @note
29  Supports only two-port devices.\n
30  @see Z0KHelper to see how the reference impedance
31  and scaling factor are determined.
32  """
33  ABCD=array(ABCD)
34  (Z0,K)=Z0KHelper((Z0,K),len(ABCD))
35  Z01=Z0.item(0,0)
36  Z02=Z0.item(1,1)
37  K1=K.item(0,0)
38  K2=K.item(1,1)
39  C11=array([[0,0],[1.0/(2.0*K2),Z02/(2.0*K2)]])
40  C12=array([[1.0/(2.0*K1),-Z01/(2.0*K1)],[0,0]])
41  C21=array([[0,0],[1.0/(2.0*K2),-Z02/(2.0*K2)]])
42  C22=array([[1.0/(2.0*K1),Z01/(2.0*K1)],[0,0]])
43  return (C21+C22.dot(ABCD)).dot(inv(C11+C12.dot(ABCD))).tolist()
