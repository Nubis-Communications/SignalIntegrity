24 class FractionalDelayFilterLinear(FirFilter):
25  """linear fractional delay filter"""
26  def __init__(self,F,accountForDelay=True):
27  """Constructor
28 
29  applies a two-tap linear interpolating filter.
30 
31  @param F float amount of delay to apply. The delay is in samples of the input waveform.
32  @param accountForDelay (optional) boolean whether to account for the delay
33  @remark
34  if accountForDelay, then the filter provides a sample phase adjustment, meaning
35  that there is no actual delay applied to the waveform, but the time axis under
36  the waveform is shifted. This is the usual way to apply this filter and is used
37  to adapt waveforms on different time axes to each other.\n
38  if not accountForDelay, then the filter actually delays waveforms by the delay
39  specified.
40  """
41  # pragma: silent exclude
42  from SignalIntegrity.Lib.TimeDomain.Filters.FilterDescriptor import FilterDescriptor
43  # pragma: include
44  FirFilter.__init__(self,FilterDescriptor(1,
45  (F if F >= 0 else 1+F) if accountForDelay else 0,1),
46  [1-F,F] if F >= 0 else [-F,1+F])
