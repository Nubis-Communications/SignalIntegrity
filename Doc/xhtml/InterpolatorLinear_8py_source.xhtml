24 class FractionalDelayFilterLinear(FirFilter):
25  """linear fractional delay filter"""
26  def __init__(self,F,accountForDelay=True):
27  """Constructor
28  applies a two-tap linear interpolating filter.
29  @param F float amount of delay to apply. The delay is in samples of the input waveform.
30  @param accountForDelay (optional) boolean whether to account for the delay
31  @remark if accountForDelay, then the filter provides a sample phase adjustment, meaning
32  that there is no actual delay applied to the waveform, but the time axis under
33  the waveform is shifted. This is the usual way to apply this filter and is used
34  to adapt waveforms on different time axes to each other.\n
35  if not accountForDelay, then the filter actually delays waveforms by the delay
36  specified.
37  """
38  # pragma: silent exclude
39  from SignalIntegrity.Lib.TimeDomain.Filters.FilterDescriptor import FilterDescriptor
40  # pragma: include
41  FirFilter.__init__(self,FilterDescriptor(1,
42  (F if F >= 0 else 1+F) if accountForDelay else 0,1),
43  [1-F,F] if F >= 0 else [-F,1+F])

54  # pragma: include

66  interpolation filter.
67  """
68  # pragma: silent exclude
69  from SignalIntegrity.Lib.TimeDomain.Waveform.Waveform import Waveform
70  # pragma: include
71  fd=self.FilterDescriptor()
72  us=[0. for k in range(len(wf)*fd.U)]
73  for k in range(len(wf)):
74  us[k*fd.U]=wf[k]
