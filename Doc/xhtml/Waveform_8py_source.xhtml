208  def ReadFromFile(self,fileName):
209  """reads a waveform from a file
210  @param fileName string name of file to read
211  @return self
212  @note this DOES affect self
213  @note the normal waveform format is one number per line starting with the
214  horizontal offset followed by the number of points followed by the sample
215  rate. The remaining lines contain one waveform point per line. This is
216  the format output by SignalIntegrity. However, if the data is all on one line, then
217  the format is assumed to be LeCroy MathPack format with the first point
218  being the number of points, and the remaining points being time and value.
219  """
220  # pragma: silent exclude
221  try:
222  # pragma: include outdent
223  with open(fileName,'rU' if sys.version_info.major < 3 else 'r') as f:
224  data=f.readlines()
225  # pragma: silent exclude
226  if len(data)==1:
227  data=data[0].split()
228  NumPts=int(float(data[0])+0.5)
229  HorOffset=float(data[1])
230  SampleRate=1./(float(data[3])-HorOffset)
231  Values=[float(data[k*2+2]) for k in range(NumPts)]
232  else:
233  # pragma: silent include outdent
234  HorOffset=float(data[0])
235  NumPts=int(float(data[1])+0.5)
236  SampleRate=float(data[2])
237  Values=[float(data[k+3]) for k in range(NumPts)]
238  # pragma: silent indent
239  self.td=TimeDescriptor(HorOffset,NumPts,SampleRate)
240  list.__init__(self,Values)
241  # pragma: silent exclude indent
242  except IOError:
243  raise SignalIntegrityExceptionWaveformFile(fileName+' not found')
244  # pragma: include
245  return self

282  def Adapt(self,td):
283  """adapts waveform to time descriptor
284 
285  Waveform adaption is performed using upsampling, decimation, fractional delay,
286  and waveform point trimming.
287 
288  @param td instance of class TimeDescriptor to adapt waveform to
289  @return instance of class Waveform containing self adapted to the time descriptor
290  @note does not affect self.
291  @note the static member variable adaptionStrategy determines how to interpolate. 'SinX' means
292  to use sinx/x interpolation, 'Linear' means to use linear interpolation.
293  @see InterpolatorSinX
294  @see SignalIntegrity.TimeDomain.Filters.InterpolatorSinX.FractionalDelayFilterSinX
295  @see SignalIntegrity.TimeDomain.Filters.InterpolatorSinX.FractionalDelayFilterSinX
296  @see SignalIntegrity.TimeDomain.Filters.InterpolatorLinear.InterpolatorLinear
297  @see SignalIntegrity.TimeDomain.Filters.InterpolatorLinear.FractionalDelayFilterLinear
298  @see SignalIntegrity.TimeDomain.Filters.WaveformTrimmer.WaveformTrimmer
299  @see SignalIntegrity.TimeDomain.Filters.WaveformDecimator.WaveformDecimator
300  @see SignalIntegrity.Rat.Rat
301 
302  """
303  # pragma: silent exclude
304  from SignalIntegrity.Lib.TimeDomain.Filters.InterpolatorSinX import InterpolatorSinX
305  from SignalIntegrity.Lib.TimeDomain.Filters.InterpolatorSinX import FractionalDelayFilterSinX
306  from SignalIntegrity.Lib.TimeDomain.Filters.InterpolatorLinear import InterpolatorLinear
307  from SignalIntegrity.Lib.TimeDomain.Filters.InterpolatorLinear import FractionalDelayFilterLinear
308  from SignalIntegrity.Lib.TimeDomain.Filters.WaveformTrimmer import WaveformTrimmer
309  from SignalIntegrity.Lib.TimeDomain.Filters.WaveformDecimator import WaveformDecimator
310  from SignalIntegrity.Lib.Rat import Rat
311  # pragma: include
312  wf=self
313  (upsampleFactor,decimationFactor)=Rat(td.Fs/wf.td.Fs)
314  if upsampleFactor>1:
315  wf=wf*(InterpolatorSinX(upsampleFactor) if wf.adaptionStrategy=='SinX'
316  else InterpolatorLinear(upsampleFactor))
317  ad=td/wf.td
318  f=ad.D-int(ad.D)
319  if not f==0.0:
320  wf=wf*(FractionalDelayFilterSinX(f,True) if wf.adaptionStrategy=='SinX'
321  else FractionalDelayFilterLinear(f,True))
322  ad=td/wf.td
323  if decimationFactor>1:
324  decimationPhase=int(round(ad.TrimLeft())) % decimationFactor
325  wf=wf*WaveformDecimator(decimationFactor,decimationPhase)
326  ad=td/wf.td
327  tr=WaveformTrimmer(max(0,int(round(ad.TrimLeft()))),
328  max(0,int(round(ad.TrimRight()))))
329  wf=wf*tr
330  return wf
331  def Measure(self,time):
332  """measures a value at a given time
333  @param time float time to measure the value at.
334  @return value at time specified
335  @note will return None if time is not within the waveform
336  @note linearly interpolates nearest point
337  """
338  sample=(time-self.td.H)*self.td.Fs
339  k=int(math.floor(sample))
340  if k < 0 or k > (self.td.K-1): return None
341  frac=sample-k
342  res=frac*(self[k+1]-self[k])+self[k]
343  return res

361  def Integral(self,c=0.,addPoint=True,scale=True):
362  """integral of waveform
363 
364  the integral is calculated using Riemann sums (as opposed to trapezoidal
365  integration.
366 
367  @param c (optional) float value to add to the integral waveform
368  @param addPoint (optional) boolean whether to add a point to the waveform before
369  the first point. the value added is c.
370  @param scale (optional) boolean whether to multiply each sum by the sample period
371  providing a true integral. Otherwise, the values are simply summed.
372  """
373  td=copy(self.td)
374  i=[0 for k in range(len(self))]
375  T=1./td.Fs if scale else 1.
376  for k in range(len(i)):
377  if k==0:
378  i[k]=self[k]*T+c
379  else:
380  i[k]=i[k-1]+self[k]*T
381  td.H=td.H+(1./2.)*(1./td.Fs)
382  if addPoint:
383  td.K=td.K+1
384  td.H=td.H=td.H-1./td.Fs
385  i=[c]+i
386  return Waveform(td,i)
387  def Derivative(self,c=0.,removePoint=True,scale=True):
388  """derivative of waveform
389 
390  the derivative is calculated using the difference divided by the sample period.
391 
392  @param c (optional) this value is superfluous and not used.
393  @param removePoint (optional) boolean whether to remove the first point. If the
394  first point is not removed, it is zero.
395  @param scale (optional) boolean whether to divide each difference by the sample period
396  providing a true derivative. Otherwise, the values are simply subtracted.
397  @todo remove argument c.
398  """
399  td=copy(self.td)
400  vl=copy(self)
401  T=1./td.Fs if scale else 1.
402  for k in range(len(vl)):
403  if k==0:
404  vl[k]=0.
405  else:
406  vl[k]=(self[k]-self[k-1])/T
407  td.H=td.H-(1./2.)*(1./td.Fs)
408  if removePoint:
409  td.K=td.K-1
410  td.H=td.H+1./td.Fs
411  vl=vl[1:]
412  return Waveform(td,vl)
