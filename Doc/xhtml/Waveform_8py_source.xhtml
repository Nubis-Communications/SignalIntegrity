103  def __add__(self,other):
104  """overloads +
105  @param other instance of class Waveform or float, int, complex to add.
106  @return instance of class Waveform with other added to self

108  @note

110

215  def ReadFromFile(self,fileName):
216  """reads a waveform from a file
217  @param fileName string name of file to read
218  @return self
219  @note this DOES affect self
220  @note the normal waveform format is one number per line starting with the
221  horizontal offset followed by the number of points followed by the sample
222  rate. The remaining lines contain one waveform point per line. This is
223  the format output by SignalIntegrity. However, if the data is all on one line, then
224  the format is assumed to be LeCroy MathPack format with the first point
225  being the number of points, and the remaining points being time and value.
226  @note if the file extension is '.trc', then LeCroy waveform format is assumed
227  """
228  # pragma: silent exclude
229  _, file_extension = os.path.splitext(fileName)
230  if file_extension == '.trc':
231  self.ReadLeCroyWaveform(fileName)
232  return self
233  try:
234  # pragma: include outdent
235  with open(fileName,'rU' if sys.version_info.major < 3 else 'r') as f:
236  data=f.readlines()
237  # pragma: silent exclude
238  if len(data)==1:
239  data=data[0].split()
240  NumPts=int(float(data[0])+0.5)
241  HorOffset=float(data[1])
242  SampleRate=1./(float(data[3])-HorOffset)
243  Values=[float(data[k*2+2]) for k in range(NumPts)]
244  else:
245  # pragma: silent include outdent
246  HorOffset=float(data[0])
247  NumPts=int(float(data[1])+0.5)
248  SampleRate=float(data[2])
249  Values=[float(data[k+3]) for k in range(NumPts)]
250  # pragma: silent indent
251  self.td=TimeDescriptor(HorOffset,NumPts,SampleRate)
252  list.__init__(self,Values)
253  # pragma: silent exclude indent
254  except IOError:
255  raise SignalIntegrityExceptionWaveformFile(fileName+' not found')
256  # pragma: include
257  return self

302  def Adapt(self,td):
303  """adapts waveform to time descriptor
304  Waveform adaption is performed using upsampling, decimation, fractional delay,
305  and waveform point trimming.
306  @param td instance of class TimeDescriptor to adapt waveform to
307  @return instance of class Waveform containing self adapted to the time descriptor
308  @note does not affect self.
309  @note the static member variable adaptionStrategy determines how to interpolate. 'SinX' means
310  to use sinx/x interpolation, 'Linear' means to use linear interpolation.
311  @see InterpolatorSinX
312  @see SignalIntegrity.TimeDomain.Filters.InterpolatorSinX.FractionalDelayFilterSinX
313  @see SignalIntegrity.TimeDomain.Filters.InterpolatorSinX.FractionalDelayFilterSinX
314  @see SignalIntegrity.TimeDomain.Filters.InterpolatorLinear.InterpolatorLinear
315  @see SignalIntegrity.TimeDomain.Filters.InterpolatorLinear.FractionalDelayFilterLinear
316  @see SignalIntegrity.TimeDomain.Filters.WaveformTrimmer.WaveformTrimmer
317  @see SignalIntegrity.TimeDomain.Filters.WaveformDecimator.WaveformDecimator
318  @see SignalIntegrity.Rat.Rat
319  """
320  # pragma: silent exclude
321  from SignalIntegrity.Lib.TimeDomain.Filters.InterpolatorSinX import InterpolatorSinX
322  from SignalIntegrity.Lib.TimeDomain.Filters.InterpolatorSinX import FractionalDelayFilterSinX
323  from SignalIntegrity.Lib.TimeDomain.Filters.InterpolatorLinear import InterpolatorLinear
324  from SignalIntegrity.Lib.TimeDomain.Filters.InterpolatorLinear import FractionalDelayFilterLinear
325  from SignalIntegrity.Lib.TimeDomain.Filters.WaveformTrimmer import WaveformTrimmer
326  from SignalIntegrity.Lib.TimeDomain.Filters.WaveformDecimator import WaveformDecimator
327  from SignalIntegrity.Lib.Rat import Rat
328  # pragma: include
329  wf=self
330  (upsampleFactor,decimationFactor)=Rat(td.Fs/wf.td.Fs)
331  if upsampleFactor>1:
332  # pragma: silent exclude
333  if wf.td.K*upsampleFactor > self.maximumWaveformSize:
334  raise SignalIntegrityExceptionWaveform('waveform too large to process')
335  # pragma: include
336  wf=wf*(InterpolatorSinX(upsampleFactor) if wf.adaptionStrategy=='SinX'
337  else InterpolatorLinear(upsampleFactor))
338  ad=td/wf.td
339  f=ad.D-int(math.floor(ad.D))
340  if not ((f<self.epsilon) or ((1-f)<self.epsilon)):
341  wf=wf*(FractionalDelayFilterSinX(f,True) if wf.adaptionStrategy=='SinX'
342  else FractionalDelayFilterLinear(f,True))
343  ad=td/wf.td
344  if decimationFactor>1:
345  decimationPhase=int(round(ad.TrimLeft())) % decimationFactor
346  wf=wf*WaveformDecimator(decimationFactor,decimationPhase)
347  ad=td/wf.td
348  tr=WaveformTrimmer(max(0,int(round(ad.TrimLeft()))),
349  max(0,int(round(ad.TrimRight()))))
350  wf=wf*tr
351  return wf
352  def Measure(self,time):
353  """measures a value at a given time
354  @param time float time to measure the value at.
355  @return value at time specified
356  @note will return None if time is not within the waveform
357  @note linearly interpolates nearest point
358  """
359  sample=(time-self.td.H)*self.td.Fs
360  k=int(math.floor(sample))
361  if k < 0 or k > (self.td.K-1): return None
362  frac=sample-k
363  res=frac*(self[k+1]-self[k])+self[k]
364  return res

382  def Integral(self,c=0.,addPoint=True,scale=True):
383  """integral of waveform
384  the integral is calculated using Riemann sums (as opposed to trapezoidal
385  integration.
386  @param c (optional) float value to add to the integral waveform
387  @param addPoint (optional) boolean whether to add a point to the waveform before
388  the first point. the value added is c.
389  @param scale (optional) boolean whether to multiply each sum by the sample period
390  providing a true integral. Otherwise, the values are simply summed.
391  """
392  td=copy(self.td)
393  i=[0 for k in range(len(self))]
394  T=1./td.Fs if scale else 1.
395  for k in range(len(i)):
396  if k==0:
397  i[k]=self[k]*T+c
398  else:
399  i[k]=i[k-1]+self[k]*T
400  td.H=td.H+(1./2.)*(1./td.Fs)
401  if addPoint:
402  td.K=td.K+1
403  td.H=td.H=td.H-1./td.Fs
404  i=[c]+i
405  return Waveform(td,i)
406  def Derivative(self,c=0.,removePoint=True,scale=True):
407  """derivative of waveform
408  the derivative is calculated using the difference divided by the sample period.
409  @param c (optional) this value is superfluous and not used.
410  @param removePoint (optional) boolean whether to remove the first point. If the
411  first point is not removed, it is zero.
412  @param scale (optional) boolean whether to divide each difference by the sample period
413  providing a true derivative. Otherwise, the values are simply subtracted.
414  @todo remove argument c.
415  """
416  td=copy(self.td)
417  vl=copy(self)
418  T=1./td.Fs if scale else 1.
419  for k in range(len(vl)):
420  if k==0:
421  vl[k]=0.
422  else:
423  vl[k]=(self[k]-self[k-1])/T
424  td.H=td.H-(1./2.)*(1./td.Fs)
425  if removePoint:
426  td.K=td.K-1
427  td.H=td.H+1./td.Fs
428  vl=vl[1:]
429  return Waveform(td,vl)
