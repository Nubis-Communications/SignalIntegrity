30 class LevMar(CallBacker):
31  """Implements the Levenberg-Marquardt algorithm for non-linear fitting
32  To use this class, you derive your class for fitting some arbitrary
33  function to some data from LeVmar. The function must contain, at a minimum, an
34  __init__ function for construction and a function fF(self,a), which implements
35  the function y=f(a) where ideally a is a vector of numbers and y is a vector of
36  numbers in matrix form. In other words, if you are fitting a 3 variable function
37  f(a0,a1,a2)=y - you are trying to find the best values of a0, a1, and a2 that
38  cause f(a0,a1,a2) to be the closes to y in a least-squares sense, then your
39  a would be something like [[a0],[a1],[a2]] and your y would look similar. In
40  this way, a is a 3x1 element matrix.
41  During the fit, LevMar will call the function fJ, which in turn calls fPartialFPartiala
42  with a and m being the index of the variable in a to take the derivative with respect
43  to. If your derived class does not overload fJ and/or fPartialFPartiala, LevMar will
44  compute a numerical derivative using the value self.m_epsilon as the delta. If
45  you have analytic derivatives and don't want derivatives calculated numerically,
46  you overload fJ and return a matrix that is R a M where R is the number of elements
47  in y and M is the number of elements in a where each J[r][m] is the partial derivative
48  of F with respect a[m] at element r of the output.
49  """
50  def __init__(self,callback=None):
51  """Constructor
52  @param callback a callback function to call during calculation
53  """
54  self.m_lambda=1000
55  self.m_lambdamin=1e-15
56  self.m_lambdamax=1e9
57  self.m_lambdaMultiplier = 2.
58  self.m_epsilon = 1e-19
59  self.ccm=FitConvergenceMgr()
60  CallBacker.__init__(self,callback)
61  def fF(self,a):
62  """implements F(a)
63  @param a list of lists representing matrix a
64  @return F(a) in the overloaded function
65  if the function is not overloaded in the derived class, raises an exception
66  @throw raise if not overloaded.
67  """
68  raise SignalIntegrityExceptionFitter('fF must be overloaded')
69  def fPartialFPartiala(self,a,m,Fa=None):
70  """partial derivative of F(a) with respect to a[m]
71  @param a list of lists matrix a
72  @param m index of element in a which to take partial derivative with respect to
73  @param Fa optional previously calculated F(a) to avoid double calculation
74  @return the partial derivative of F(a) with respect to element m in a
75  """
76  aplusDeltaa = copy.copy(a)
77  aplusDeltaa[m][0]=aplusDeltaa[m][0]+self.m_epsilon
78  if Fa is None:
79  Fa = self.fF(a)
80  dFa = (self.fF(aplusDeltaa)-Fa)/self.m_epsilon
81  return dFa
82  def fJ(self,a,Fa=None):
83  """Calculates the Jacobian matrix.
84  @param a list of lists matrix a
85  @param Fa optional previously calculated F(a) to avoid double calculation
86  @return the Jacobian matrix
87  The Jacobian matrix is R x M where R is the number of elements in F(a) and M
88  is the number of elements in a where each J[r][m] is the partial derivative
89  of F with respect a[m] at element r of the output.
90 
91  This function can be overloaded in the derived class if analytic partial derivatives are
92  known, otherwise numerical partial derivatives are calculated using fPartialFPartiala()
93  """
94  if Fa is None:
95  Fa=self.fF(a)
96  M = len(a)
97  R = len(Fa)
98  J = zeros((R,M))
99  for m in range(M):
100  pFpam=self.fPartialFPartiala(a,m,Fa)
101  for r in range(R):
102  J[r][m]=pFpam[r][0]
103  return J
104  def AdjustVariablesAfterIteration(self,a):
105  """AdjustVariablesAfterIteration
106  @param a list of lists matrix a
107  a contains the new variables produced as a result of an iteration
108 
109  if not overridden, nothing happens, but if this function is overridden in the derived class,
110  it provides an opportunity to make adjustments to the variables. Examples are things like making
111  variables real when during iteration they could become complex or making variables positive.
112  The good thing about this function is that these adjustments are made prior to making decisions
113  on the success of the iteration meaning that sometime the a resulting from an iteration reduces
114  mean-squared error between F(a) and y, but the a values are invalid and only valid after adjustment.
115  If, after the adjustment, the mean-squared error is not reduced, the iteration fails.
116  """
117  return a
118  def Initialize(self,a,y,w=None):
119  """Initializes the fitter for optimizing the values of F(a) such that they equal y in a weighted
120  least-squares sense, with w providing the weights.
121  @param a list of lists matrix a
122  @param y list of lists matrix y
123  @param w (optional) list of lists weights matrix w
124  """
125  self.m_a = copy.copy(array(a))
126  self.m_y = copy.copy(array(y))
127  if w is None:
128  self.m_sumw=len(y)
129  self.m_W=1.0
130  else:
131  self.m_W = diag(w)
132  self.m_sumw = 0
133  for r in range(w.rows()):
134  self.m_sumw = self.m_sumw + w[r][0]
135  self.m_Fa = self.fF(self.m_a)
136  self.m_r=self.m_Fa-self.m_y
137  self.m_mse=math.sqrt(self.m_r.conj().T.dot(self.m_W).dot(
138  self.m_r)[0][0].real/self.m_sumw)
139  self.m_lambdaTracking = [self.m_lambda]
140  self.m_mseTracking = [self.m_mse]
141  self.m_J = None
142  self.m_H = None
143  self.m_JHWr = None
144  self.ccm=FitConvergenceMgr()
145  def Iterate(self):
146  """Performs one iteration.
147  Usually, this function should not be used. It can be used to take each iteration manually.
148  Usually Solve() is used, which takes these iterations, but tests convergence and decides
149  when the iterating should end.
150  """
151  if self.m_Fa is None:
152  self.m_Fa=self.fF(self.m_a)
153  if self.m_r is None:
154  self.m_r=self.m_Fa-self.m_y
155  if self.m_J is None:
156  self.m_J=self.fJ(self.m_a,self.m_Fa)
157  # pragma: silent exclude
158  if self.m_J.shape[1]>self.m_J.shape[0]: # fat matrix - underconstrained
159  if self.m_H is None:
160  self.m_H=lstsq(self.m_J,self.m_r,0.001)[0]
161  Deltaa=self.m_H/float(self.m_lambda)
162  else:
163  # pragma: include outdent
164  if self.m_H is None:
165  self.m_H=self.m_J.conj().T.dot(self.m_W).dot(self.m_J)
166  if self.m_JHWr is None:
167  self.m_JHWr=self.m_J.conj().T.dot(self.m_W).dot(self.m_r)
168  HplD=copy.copy(self.m_H)
169  for r in range(HplD.shape[0]): HplD[r][r]*=(1.+self.m_lambda)
170  # pragma: silent exclude
171  try:
172  # pragma: include outdent
173  Deltaa=solve(HplD,self.m_JHWr)
174  # pragma: silent exclude indent
175  except LinAlgError:
176  Deltaa=lstsq(HplD,self.m_JHWr,0.001)[0]
177  # pragma: include indent
178  newa=self.m_a-Deltaa
179  newa=self.AdjustVariablesAfterIteration(newa)
180  newFa = self.fF(newa)
181  newr=newFa-self.m_y
182  newmse=math.sqrt(newr.conj().T.dot(self.m_W).dot(newr)[0][0].real/self.m_sumw)
183  if newmse < self.m_mse:
184  self.m_mse = newmse
185  self.m_a = newa
186  self.m_lambda = max(self.m_lambda/
187  self.m_lambdaMultiplier,self.m_lambdamin)
188  self.m_Fa = newFa
189  self.m_r = newr
190  self.m_J = None
191  self.m_H = None
192  self.m_JHWr = None
193  else:
194  self.m_lambda = min(self.m_lambda*
195  self.m_lambdaMultiplier,self.m_lambdamax)
196  self.ccm.IterationResults(self.m_mse, self.m_lambda)
197  self.m_lambdaTracking.append(self.m_lambda)
198  self.m_mseTracking.append(self.m_mse)
199  def Solve(self):
200  """Solves for a such that F(a) equals y in a weighted least-squares sense."""
201  self.Iterate()
202  self.m_lastMse=self.m_mse
203  self.m_MseAcc=self.m_lastMse
204  while self.ccm.Continue():
205  self.CallBack(self.ccm._IterationsTaken)
206  self.Iterate()
207  return self
