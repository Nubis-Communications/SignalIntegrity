46  """Calculates and stores internally a symbolic representation

48  @return self
49  @see Symbolic
50  """
51  self.Check()
52  self._LaTeXSi()
53  numMeas=len(self.pMeasurementList)
54  vemsi = MatrixMultiply(
55  self.VoltageExtractionMatrix(self.pMeasurementList), self.SIPrime(True))
56  oneElementVemsi = False
57  if len(vemsi) == 1:
58  if len(vemsi[0]) == 1: oneElementVemsi = True
59  vemsi = Matrix2LaTeX(vemsi, self._SmallMatrix())
60  veosi = MatrixMultiply(
61  self.VoltageExtractionMatrix(self.pOutputList), self.SIPrime(True))
62  oneElementVeosi = False
63  if len(veosi) == 1:
64  if len(veosi[0]) == 1: oneElementVeosi = True
65  veosi = Matrix2LaTeX(veosi, self._SmallMatrix())
66  if self.pStimDef is None:
67  numDeg=len(self.SIPrime(True)[0])
68  inverse='^{-1}' if numDeg==numMeas else '^\\dagger'
69  if oneElementVemsi and oneElementVeosi:
70  line = lfrac(' '+veosi+' ',' '+vemsi+' ')+' '
71  elif oneElementVemsi and not oneElementVeosi:
72  line = veosi+' '+lfrac('1',' '+vemsi+' ')+' '
73  elif not oneElementVemsi and oneElementVeosi:
74  line = '\\left( '+veosi+'\\right)\\cdot '+vemsi+inverse
75  elif not oneElementVemsi and not oneElementVeosi:
76  line = veosi+'\\cdot '+vemsi+inverse
77  else:
78  numDeg=len(self.pStimDef[0])
79  inverse='^{-1}' if numDeg==numMeas else '^\\dagger'

82  fvemsi='\\left( '+vemsi+'\\right)' if oneElementVemsi else vemsi
83  line = '\\left[ '+fveosi+'\\cdot '+D+'\\right]\\cdot'+\

85  if len(self.pMeasurementList) == 1 and len(self.pOutputList) == 1: H = 'H'
86  else: H = '\\mathbf{H}'
87  return self._AddEq(H+' = '+line)
