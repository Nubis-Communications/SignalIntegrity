274  def ImageRMSE(imageA, imageB):
275  import numpy as np
276  import math
277 
278  imageA=np.array(imageA)
279  imageB=np.array(imageB)
280  # the 'Mean Squared Error' between the two images is the
281  # sum of the squared difference between the two images;
282  # NOTE: the two images must have the same dimension
283  err = np.sum((imageA.astype("float") - imageB.astype("float")) ** 2)
284  err /= float(imageA.shape[0] * imageA.shape[1])
285  # return the MSE, the lower the error, the more "similar"
286  # the two images are
287  return math.sqrt(err)

327  def are_jsons_approx_equal(js1: JsonType, js2: JsonType, precision: int) -> bool:
328  """Compares two json objects and returns True if these are approximately equal.
329 
330  Float values are rounded to the significant figures specified as `precision`
331  when being compared.
332 
333  Args:
334  js1: json object to compare
335  js2: json object to compare
336  precision: significant figures applied to float values
337 
338  Returns:
339  True if two json objects are approximately equal.
340 
341  Raises:
342  TypeError: if `js1` or `js2` is not a valid json object.
343  """
344  def _float_to_str_in_json(js,stringList):
345  if type(js) in (str, int, bool, type(None)):
346  jsstring=str(js)
347  #print(jsstring)
348  stringList.append(jsstring+'\n')
349  return jsstring
350  if type(js) is float:
351  if js<1e-30:
352  js=0.0
353  if js<1e-10:
354  precisionToUse=max(0,precision-2)
355  elif js<1e-4:
356  precisionToUse=max(0,precision-1)
357  else:
358  precisionToUse=precision
359  jsstring=f"{js:.{precisionToUse}e}"
360  #print(jsstring)
361  stringList.append(jsstring+'\n')
362  return jsstring
363  if type(js) is list:
364  return [_float_to_str_in_json(x,stringList) for x in js]
365  if type(js) is dict:
366  if any([type(key) is not str for key in js.keys()]):
367  raise TypeError("One of key types is not 'str'.")
368  return {key: _float_to_str_in_json(val,stringList) for key, val in js.items()}
369  raise TypeError(f"Value of type '{type(js)}' is not a valid json object.")
370 
371  stringList1=[]
372  new_js1 = _float_to_str_in_json(js1,stringList1)

374  stringList2=[]
