80  def Waveform(self):
81  import SignalIntegrity.Lib as si
82  wfTypeProperty=self['wftype']
83  if wfTypeProperty is None:
84  waveform = None
85  else:
86  if hasattr(self, 'Wf'):
87  if self.Wf != None:
88  return self.Wf
89  wfType=wfTypeProperty.GetValue()
90  if wfType is None:
91  waveform = None
92  elif wfType=='file':
93  fileName = self['wffile'].PropertyString(stype='raw')
94  ext=str.lower(fileName).split('.')[-1]
95  if ext == 'si':
96  from SignalIntegrity.App.SignalIntegrityAppHeadless import ProjectWaveform
97  waveform=ProjectWaveform(fileName,self['wfprojname'].GetValue())
98  if waveform is None:
99  raise si.SignalIntegrityExceptionWaveform('project file: '+fileName+' could not produce waveform: '+self['wfprojname'].GetValue())
100  else:
101  waveform = si.td.wf.Waveform().ReadFromFile(fileName)
102  elif wfType == 'step':
103  amplitude=float(self['a'].GetValue())
104  startTime=float(self['t0'].GetValue())
105  risetime=float(self['rt'].GetValue())
106  waveform = si.td.wf.StepWaveform(self.WaveformTimeDescriptor(),amplitude,startTime,risetime)
107  elif wfType == 'pulse':
108  amplitude=float(self['a'].GetValue())
109  startTime=float(self['t0'].GetValue())
110  pulseWidth=float(self['w'].GetValue())
111  risetime=float(self['rt'].GetValue())
112  waveform = si.td.wf.PulseWaveform(self.WaveformTimeDescriptor(),amplitude,startTime,pulseWidth,risetime)
113  elif wfType == 'prbs':
114  polynomial=int(self['prbs'].GetValue())
115  bitrate=float(self['br'].GetValue())
116  risetime=float(self['rt'].GetValue())
117  amplitude=float(self['a'].GetValue())
118  delay=float(self['t0'].GetValue())
119  waveform = si.prbs.PseudoRandomWaveform(polynomial,bitrate,amplitude,risetime,delay,self.WaveformTimeDescriptor())
120  elif wfType == 'mlw':
121  polynomial=int(self['prbs'].GetValue())
122  baudrate=float(self['br'].GetValue())
123  bitspersymbol=int(self['bps'].GetValue())
124  risetime=float(self['rt'].GetValue())
125  amplitude=float(self['a'].GetValue())
126  delay=float(self['t0'].GetValue())
127  waveform = si.prbs.MultiLevelWaveform(polynomial,baudrate,bitspersymbol,amplitude,risetime,delay,self.WaveformTimeDescriptor())
128  elif wfType == 'clock':
129  clockrate=float(self['f'].GetValue())
130  risetime=float(self['rt'].GetValue())
131  amplitude=float(self['a'].GetValue())
132  delay=float(self['t0'].GetValue())
133  waveform = si.prbs.ClockWaveform(clockrate,amplitude,risetime,delay,self.WaveformTimeDescriptor())
134  elif wfType == 'sine':
135  amplitude=float(self['a'].GetValue())
136  frequency=float(self['f'].GetValue())
137  phase=float(self['ph'].GetValue())
138  start=float(self['t0'].GetValue())
139  stop=float(self['tf'].GetValue())
140  waveform = si.td.wf.SineWaveform(self.WaveformTimeDescriptor(),amplitude,frequency,phase,start,stop)
141  elif wfType == 'noise':
142  sigma=float(self['vrms'].GetValue())
143  waveform = si.td.wf.NoiseWaveform(self.WaveformTimeDescriptor(),sigma)
144  elif wfType == 'networkanalyzerport':
145  td=self.WaveformTimeDescriptor()
146  if self['state']['Value'] == 'off':
147  waveform = si.td.wf.Waveform(td)
148  else:
149  if self['st']['Value'] == 'CW':
150  waveform = si.td.wf.Waveform(td)
151  dbm=float(self['pow']['Value'])
152  waveform[td.IndexOfTime(0.0)]=pow(10.,(float(self['pow']['Value'])-13.010)/20.0)*math.sqrt(td.K)
153  elif self['st']['Value'] == 'TDRImpulse':
154  waveform = si.td.wf.StepWaveform(td,2.*float(self['ia']['Value'])/td.Fs,0.0,float(self['rt']['Value'])).Derivative(removePoint=True,scale=True)
155  elif self['st']['Value'] == 'TDRStep':
156  waveform = si.td.wf.StepWaveform(td,2.*float(self['a']['Value']),0.0,float(self['rt']['Value']))
157  elif wfType == 'DC':
158  amplitude=float(self['a'].GetValue())
159  waveform=amplitude
160  return waveform
