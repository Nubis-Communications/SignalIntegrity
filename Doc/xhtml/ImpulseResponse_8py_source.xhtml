50  def FrequencyResponse(self,fd=None,adjustLength=True):
51  """produces the frequency response
52 
53  impulse responses and frequency responses are equivalent and can be converted back
54  and forth into each other. This method converts the impulse response to its corresponding
55  frequency domain equivalent.
56 
57  @attention users of this function should only either supply the fd argument or not. Not
58  supplying the argument provides the generic frequency response associated with this impulse
59  response such that self.FrequencyResponse().ImpulseResponse() provides the same answer.
60 
61  Supplying fd supplies the resulting frequency response resampled onto another frequency scale.
62 
63  @param fd (optional) instance of class FrequencyList (defaults to None).
64  @param adjustLength (optional) bool whether to adjust the length. (defaults to True).
65 
66  @note All impulse responses are evenly spaced
67 
68  @note whether a frequency descriptor is specified and whether to adjust length determines
69  all possibilities of what can happen:
70 
71  | fd | adjustLength | Generates: |
72  |:---------:|:------------:|:---------------------------------------:|
73  | None | False | generic frequency response |
74  | None | True | frequency response with length adjusted |
75  | provided | don't care | CZT resamples to fd (adjusts length) |
76 
77  The frequency descriptor, if provided, provides the frequency points to resample to, otherwise
78  the frequency descriptor associated with the internal time descriptor inherent to the impulse
79  response is used.
80 
81  Length adjustment means that although the impulse response may start at time zero, or some
82  other time, the assumption is that there are an equal number of points for negative and positve
83  time and that time zero is in the center of these points.
84 
85  This assumption for length adjustment is what allows an impulse response to be filled in with
86  positive only times, but allow an equality to exist between all frequency responses and impulse
87  responses.
88 
89  Basically, the time adjustment can be seen as calculating the number of points before time zero
90  and the number of points after time zero and calculating the total number of points in the waveform
91  (for the purposes of frequency response calculation) as the max of these two numbers multiplied by two.
92  """
93  # pragma: silent exclude
94  from SignalIntegrity.Lib.FrequencyDomain.FrequencyResponse import FrequencyResponse
95  # pragma: include
96  if not fd and not adjustLength:
97  X=fft.fft(self.Values())
98  fd=self.td.FrequencyList()
99  return FrequencyResponse(fd,[X[n] for n in range(fd.N+1)]).\
100  _DelayBy(self.td.H)
101  if not fd and adjustLength:
102  return self._AdjustLength().FrequencyResponse(None,adjustLength=False)
103  if fd:
104  return self.FrequencyResponse().Resample(fd)
105  def _AdjustLength(self):
106  """adjusts the length of the impulse response
107 
108  Calculates the number of points in the impulse response as twice the maximimum of the positive and
109  negative time-domain points.
110 
111  @return instance of class ImpulseResponse with the impulse centered at time zero, an even number of
112  points, and an equal number of points before and after time zero.
113  """
114  td = self.td
115  PositivePoints = int(max(0,math.floor(td.H*td.Fs+td.K+0.5)))
116  NegativePoints = int(max(0,math.floor(-td.H*td.Fs+0.5)))
117  P=max(PositivePoints,NegativePoints)*2
118  return self._Pad(P)
119  def _Pad(self,P):
120  """Pads the impulse response
121 
122  The impulse response is padded to P points with an equal number of points before and after
123  time zero.
124 
125  @param P integer number of points to pad the impulse response to.
126  @return an instance of class ImpulseResponse with the zero padded points.
127  @attention P must be even - not checked - it must add equal points to the left
128  and right of the impulse response.
129  @note
130  if K is the number of points in the selfs frequency response then:
131 
132  | P and K | outcome
133  |:--------:|:--------------------------------------------:|
134  | P==K | the original response is returned |
135  | P<K | the response is truncated to P time points |
136  | P>K | the response is zero padded to P time points |
137  """
138  K=len(self)
139  if P==K: x = self.Values()
140  elif P<K: x=[self[k] for k in range((K-P)//2,K-(K-P)//2)]
141  else:
142  x=[0 for p in range((P-K)//2)]
143  x=x+self.Values()+x
144  td = self.td
145  return ImpulseResponse(TimeDescriptor(td.H-(P-K)/2./td.Fs,P,td.Fs),x)
146  def _FractionalDelayTime(self):

167  def TrimToThreshold(self,threshold):
168  """truncates an impulse response, keeping only the values above or equal to specified threshold.
169 
170  This is useful for reducing computational complexity in processing.
171 
172  @param threshold float threshold to apply to the values in the waveform.
173  @attention the threshold specified is a fraction of the maximum absolute value of the values
174  in the waveform.
175  @note that absolute values are utilized in value/threshold comparison.
176  @note the trimming is performed by finding the lowest and highest time value above or equal
177  to the threshold - all values between these times are retained.
178  """
179  x=self.Values()
180  td=self.td
181  maxabsx=max(self.Values('abs'))
182  minv=maxabsx*threshold
183  for k in range(len(x)):
184  if abs(x[k]) >= minv:
185  startidx = k
186  break
187  for k in range(len(x)):
188  ki = len(x)-1-k
189  if abs(x[ki]) >= minv:
190  endidx = ki
191  break
192  if (endidx-startidx+1)//2*2 != endidx-startidx+1:
193  # the result would not have an even number of points
194  if endidx < len(x)-1:
195  # keep a point at the end if possible
196  endidx = endidx + 1
197  elif startidx > 0:
198  # keep a point at the beginning if possible
199  startidx = startidx - 1
200  else:
201  # append a zero to the end and calculate number of
202  # points with endidx+1
203  return ImpulseResponse(TimeDescriptor(td[startidx],
204  (endidx+1)-startidx+1,td.Fs),
205  [x[k] for k in range(startidx,endidx+1)]+[0.])
206  return ImpulseResponse(TimeDescriptor(td[startidx],
207  endidx-startidx+1,td.Fs),
208  [x[k] for k in range(startidx,endidx+1)])
