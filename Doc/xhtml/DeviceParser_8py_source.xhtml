24 class ParserDevice(object):
25  """class defining how to parse devices"""
26  def __init__(self,devicename,ports,arginname,defaults,frequencyDependent,func):
27  """Constructor
28  @param devicename string name of device
29  @param ports integer or string number of ports in device or None. If a string, then the string
30  is either a comma separated list of possible numbers of ports or is two numbers separated
31  by a - indicating a range of numbers of ports. If None, then the device can have
32  any number of ports.
33  @param arginname boolean whether the first argument requires a keyword or not. Simple
34  things like resistors and capacitors just have their value after the device and ports
35  declaration. This is False if the device has no arguments.
36  @param defaults dictionary containing keyword/default value pairs. if arginname is True
37  then one arguments keyword will be the empty string ''.
38  @param frequencyDependent boolean whether the device is frequency dependent.
39  @param func string that when evaluated will be the device.
40  """
41  self.devicename=devicename
42  self.ports=ports
43  self.arginname=arginname
44  defaults['z0']=50.
45  self.defaults=defaults
46  self.frequencyDependent=frequencyDependent
47  self.func=func

260  def MakeDevice(self,ports,callback, argsList, f, Z0=50.):
261  """makes a device from a set of arguments
262  The device is assigned to self.dev and self.frequencyDependent determines whether the
263  device is frequency dependent. Frequency dependent devices are assumed to be instances
264  of the class SParameters. Otherwise, they are list of list matrices.
265 
266  @param ports integer number of ports
267  @param argsList list of arguments. The name of the device is the first argument.
268  If the device has no keyword for the argument, then that argument is next. Otherwise, besides
269  the name and the argument with no keyword, the remaining arguments come in keyword/value pairs where the
270  keyword is a string and the value is the value of the keyword.
271  @param f list of frequencies
272  @param Z0 float (optional, defaults to 50.) reference impedance for the calculation
273  @return boolean whether the device was created.
274  @throw SignalIntegrityExceptionDeviceParser if the device cannot be created.
275  """
