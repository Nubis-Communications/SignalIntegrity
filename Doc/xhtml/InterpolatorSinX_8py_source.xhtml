26 def SinX(S,U,F):
27  """calculates the sinX/X filter taps for Sinx filters
28  @param S integer side samples for filter (without upsampling)
29  @param U integer upsample factor
30  @param F float fractional delay
31  @remark The filter is 2*S*U+1 in length, meaning it has S*U samples on each side of a center
32  sample point.
33  """
34  sl=[1. if float(k)/U-F-S==0 else
35  math.sin(math.pi*(float(k)/U-F-S))/(math.pi*(float(k)/U-F-S))*\
36  (1./2.+1./2.*math.cos(math.pi*(float(k)/U-S)/S))
37  for k in range(2*U*S+1)]
38  s=sum(sl)/U
39  sl=[sle/s for sle in sl]
40  return sl
