21 class MixedModeTLine(SParameters):
22  """s-parameters of a mixed-mode, balanced transmission line"""
23  def __init__(self,f,Zd,Td,Zc,Tc,Z0=50.):
24  """Constructor
25  The port numbering is ports 1 and 2 are the plus and minus terminals on
26  one side of the line and ports 3 and 4 are the plus and minus terminals
27  on the other side of the line.
28 
29  @param f list of float frequencies
30  @param Zd float or complex differential mode impedance
31  @param Td float differential (or odd-mode) electrical length (the
32  differential-mode propagation time.
33  @param Zc float or complex common-mode impedance
34  @param Tc float common (or even-mode) electrical length (the common-mode
35  propagation time.
36  @param Z0 (optional) float or complex reference impedance (defaults to 50 Ohms).
37  @note The differential mode impedance is twice the odd-mode impedance.\n
38  The common-mode impedance is half the even-mode impedance.\n
39  @note The model is appropriate for a balanced transmission line.\n
40  @note The s-parameters provided are single-ended.
41  @note This device builds a model internally and only solves the model
42  on each access to __getitem__().
43  """
44  import SignalIntegrity.Lib.SParameters.Devices as dev
45  from SignalIntegrity.Lib.Devices.MixedModeConverter import MixedModeConverter
46  from SignalIntegrity.Lib.SystemDescriptions.SystemSParametersNumeric import SystemSParametersNumeric
47  self.m_sspn=SystemSParametersNumeric()
48  self.m_spdl=[]
49  self.m_sspn.AddDevice('MM1',4,MixedModeConverter())
50  self.m_sspn.AddDevice('MM2',4,MixedModeConverter())
51  self.m_sspn.AddDevice('D',2), self.m_spdl.append(('D',dev.TLineLossless(f,2,Zd/2.,Td,Z0)))
52  self.m_sspn.AddDevice('C',2), self.m_spdl.append(('C',dev.TLineLossless(f,2,Zc*2.,Tc,Z0)))
53  self.m_sspn.ConnectDevicePort('MM1',3,'D',1)
54  self.m_sspn.ConnectDevicePort('MM2',3,'D',2)
55  self.m_sspn.ConnectDevicePort('MM1',4,'C',1)
56  self.m_sspn.ConnectDevicePort('MM2',4,'C',2)
57  self.m_sspn.AddPort('MM1',1,1)
58  self.m_sspn.AddPort('MM1',2,2)
59  self.m_sspn.AddPort('MM2',1,3)
60  self.m_sspn.AddPort('MM2',2,4)
61  SParameters.__init__(self,f,None,Z0)
62  def __getitem__(self,n):
63  """overloads [n]
64  @return list of list s-parameter matrix for the nth frequency element
65  """
66  for ds in self.m_spdl: self.m_sspn.AssignSParameters(ds[0],ds[1][n])
67  return self.m_sspn.SParameters()
