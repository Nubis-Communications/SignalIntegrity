23 class CTLE(SParameters):
24  """Continuous time linear equalizer"""
25  def __init__(self,f,gDC,gDC2,fz,fLF,fp1,fp2,Z0=50.):
26  """Constructor
27  @param f list of float frequencies
28  @param gDC float DC gain in dB
29  @param gDC2 float DC gain at peaking point in dB
30  @param fz float frequency of zero
31  @param fLF float LF frequency
32  @param fp1 float first pole frequency
33  @param fp2 float second pole frequency
34  @param Z0 float (optional, defaults to 50) reference impedance
35  """
36  self.powgDC=pow(10.,gDC/20.); self.powgDC2=pow(10.,gDC2/20.)
37  self.joverfz=1j/fz; self.joverfLF=1j/fLF; self.joverfp1=1j/fp1; self.joverfp2=1j/fp2
38  SParameters.__init__(self,f,None,Z0)
39  def _Evaluate(self,f):
40  """Evaluate CTLE
41  @param f float frequency
42  @return CTLE evaluated at f
43  """
44  N=(self.powgDC+self.joverfz*f)*(self.powgDC2+self.joverfLF*f)
45  D=(1.+self.joverfp1*f)*(1.+self.joverfp2*f)*(1.+self.joverfLF*f)
46  return N/D
47  def __getitem__(self,n):
48  return [[1.,0],[2.*self._Evaluate(self.m_f[n]),-1.]]

50 class FFE(SParameters):
51  """Linear feed-forward equalizer"""
52  def __init__(self,f,Td,taps,numPreCursorTaps,Z0=50.):
53  """Constructor
54  @param f list of float frequencies
55  @param Td float time delay of each tap
56  @param taps list of float values of the taps
57  @param numPreCursorTaps integer number of precursor taps
58  @param Z0 float (optional, defaults to 50) reference impedance
59  @remark Usually Td is the reciprocal of the baud rate or sometimes, half the reciprocal.
60  """
61  self.Td=Td
62  self.taps=taps
63  self.PC=numPreCursorTaps
64  SParameters.__init__(self,f,None,Z0)
65  def _Evaluate(self,f):
66  """Evaluate FFE
67  @param f float frequency
68  @return FFE evaluated at f
69  """
70  K=len(self.taps)
71  return sum([self.taps[k]*cmath.exp(1j*2.*cmath.pi*f*-(k-self.PC)*self.Td) for k in range(K)])
72  def __getitem__(self,n):
73  return [[1.,0],[2.*self._Evaluate(self.m_f[n]),-1.]]
