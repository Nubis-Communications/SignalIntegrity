26 class ImpedanceProfileWaveform(Waveform):
27  """Computes the impedance profile waveform from a set of s-parameters using the port specified.
28  method is 'exact','estimated' or 'approximate':
29  - 'exact' specifies to use the DFT method to deembed sections of impedance found
30  at an interface. This method is exact in simulation.
31  - 'estimated' and 'approximate' both use the step response of the system computed
32  from the s-parameters. 'estimated' calculates the Z exactly, assuming the step
33  response contains rho (the reflection coefficient), which is an estimate because
34  rho is polluted by multiple reflections.
35  - 'approximate' calculates Z using a simple
36  offset and scaling like would be employed when viewing a TDR waveform.
37  There is no reason to use 'approximate' - except for educational purposes.
38  'exact' can be quite unstable. 'Estimated' is actually usually the best method.
39 
40  align is either 'middle' or 'interface':
41  - 'middle' means that the impedance for each point produced is at the corresponding time
42  representing the the middle of a transmission line section.
43  - 'interface' means that the time of the impedance is the time of the left edge of a
44  transmission line with the corresponding impedance.
45 
46  includePortZ is set to True if you want the first point to be the impedance of the
47  port used to take the measurement.
48  """
49  rhoLimit=0.99
50  ZLimit=10e3
51  def __init__(self,sp,port=1,method='exact',align='middle',includePortZ=True,
52  adjustForDelay=True):
53  """Constructor
54  @param sp instance of class SParameters of the device
55  @param port (optional) integer 1 based port number (defaults to port 1)
56  @param method (optional) string method for computation (defaults to 'exact')
57  @param align (optional) string alignment of impedancance in waveform (defaults to 'middle')
58  @param includePortZ (optional) boolean whether to put the port reference impedance as the first point. (defaults to True)
59  @param adjustForDelay (optional) boolean whether to adjust for the delay in the impulse response (defaults to True)
60  @remark computation methods include:
61  'exact' (default) - calculates using reflection coefficient of first point computed from DFT and deembedding.
62  (this method takes longer and can diverge due to buildup of numerical inaccuracies.)
63  'estimated' - calculates the reflection coefficients directly from the step response.
64  'approximate' - calculates the reflection coefficients from the step response using an approximation.
65  @remark alignment methods include:
66  'middle' (default) - shows the impedance in the middle of the sample period distance along the line.
67  'front' - shows the impedance measured at the front of the line.
68  """
69  tdsp=sp.m_f.TimeDescriptor()
70  # assumes middle and no portZ
71  tdip=TimeDescriptor(1./(tdsp.Fs*4),tdsp.K/2,tdsp.Fs*2)
72  if not align == 'middle':
73  tdip.H=tdip.H-1./(tdsp.Fs*4)
74  if method == 'exact':
75  ip=ImpedanceProfile(sp,tdip.K,port)
76  Z=ip.Z()
77  delayAdjust=ip.m_fracD
78  elif method == 'estimated' or method == 'approximate':
79  fr=sp.FrequencyResponse(port,port)
80  rho=fr.ImpulseResponse().Integral(addPoint=True,scale=False)
81  delayAdjust=fr._FractionalDelayTime()
82  finished=False
83  for m in range(len(rho)):
84  if finished:
85  rho[m]=rho[m-1]
86  continue
87  rho[m]=max(-self.rhoLimit,min(self.rhoLimit,rho[m]))
88  if abs(rho[m])==self.rhoLimit:
89  finished=True
90  if method == 'estimated':
91  Z=[max(0.,min(sp.m_Z0*(1+rho[tdsp.K//2+1+k])/
92  (1-rho[tdsp.K//2+1+k]),self.ZLimit)) for k in range(tdip.K)]
93  else:
94  Z=[max(0.,min(sp.m_Z0+2*sp.m_Z0*rho[tdsp.K//2+1+k],self.ZLimit))
95  for k in range(tdip.K)]
96  if includePortZ:
97  tdip.H=tdip.H-1./(tdsp.Fs*2)
98  tdip.K=tdip.K+1
99  Z=[sp.m_Z0]+Z
100  if adjustForDelay: tdip.H=tdip.H+delayAdjust/2
101  Waveform.__init__(self,tdip,Z)
