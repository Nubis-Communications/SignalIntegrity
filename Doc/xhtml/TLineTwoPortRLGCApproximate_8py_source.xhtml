23 class TLineTwoPortRLGCApproximate(SParameters):
24  """s-parameters of two port RLGC (telegrapher's) transmission line
25  calculated by approximating distributed parameters with a finite number
26  of sections specified."""
27  rtFraction=.01
28  def __init__(self,f, R, Rse, L, G, C, df, Z0=50., K=0, scale=1.):
29  """Constructor
30  @param f list of float frequencies
31  @param R float DC series resistance (ohms)
32  @param Rse float series skin-effect resistance (ohms/sqrt(Hz))
33  @param L float series inductance (H)
34  @param G float DC conductance to ground (S)
35  @param C float capacitance to ground (F)
36  @param df float dissipation factor (loss-tangent) of capacitance to ground
37  @param Z0 (optional) float reference impedance (defaults to 50 ohms)
38  @param K (optional) integer number of sections (defaults to zero)
39  @param scale (optional) float amount to scale the line by (assuming all other values are per unit)
40  @note If K=0 is specified, it is modified to a value that will provided a very good numerical
41  approximation.
42 
43  The calculation is such that round-trip propagation time (twice the electrical length)
44  of any one small section is no more than one percent of the fastest possible risetime.
45  """
46  R=R*scale; Rse=Rse*scale; L=L*scale; G=G*scale; C=C*scale; df=df
47  K=int(K*scale+0.5)
48  if K==0:
49  """max possible electrical length"""
50  Td=math.sqrt(L*C)
51  Rt=0.45/f[-1] # fastest risetime
52  """sections such that fraction of risetime less than round trip
53  electrical length of one section"""
54  K=int(math.ceil(Td*2/(Rt*self.rtFraction)))
55 
56  self.m_K=K
