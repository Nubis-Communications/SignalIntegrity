29  @param f list of float frequencies
30  @param filename string name of s-parameter file or project that produces s-parameters

32  @param Z0 float (optional, defaults to None) reference impedance
33  @param **kwargs dict (optional, defaults to {}) dictionary of arguments for the file
34  @note fitting is not performed until an item from the fitted model is requested
35  """

39  self.spfile=filename
40  self.RLGC=None
41  self.kwargs=kwargs
42  def Fit(self):
43  """Fits a two-port RLGC model for the specified s-parameter file
44  @see see RLGC
45  """
46  sp=SParameterFile(self.spfile,self.Z0,**self.kwargs)
47  stepResponse=sp.FrequencyResponse(2,1).ImpulseResponse().Integral()
48  threshold=(stepResponse[len(stepResponse)-1]+stepResponse[0])/2.0
49  for k in range(len(stepResponse)):
50  if stepResponse[k]>threshold: break
51  dly=stepResponse.Times()[k]
52  rho=sp.FrequencyResponse(1,1).ImpulseResponse().Integral(scale=False).Measure(dly)
53  Z0=sp.m_Z0*(1.+rho)/(1.-rho)
54  L=dly*Z0; C=dly/Z0; guess=[0.,L,0.,C,0.,0.]
55  fitter=RLGCFitter(sp,guess)
56  (R,L,G,C,Rse,df)=fitter.Solve().Results()
57 # print "series resistance: "+ToSI(R,'ohm')
58 # print "series inductance: "+ToSI(L,'H')
59 # print "shunt conductance: "+ToSI(G,'S')
60 # print "shunt capacitance: "+ToSI(C,'F')
61 # print "skin-effect resistance: "+ToSI(Rse,'ohm/sqrt(Hz)')
62 # print "dissipation factor: "+ToSI(df,'')
63  s=self.scale
64  self.RLGC=TLineTwoPortRLGCAnalytic(self.m_f, R*s, Rse*s, L*s, G*s, C*s, df, Z0=50.)
65  return self.RLGC
