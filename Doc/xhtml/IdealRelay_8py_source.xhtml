22 def IdealRelay(ports,position,termination='open',Z0=50.):
23  """IdealRelay
24  An ideal relay with a given number of ports, where the common is always the last port.
25  I.e. for a given number ports P, port P is always the common connection and ports 1 through
26  P-1 are the throws of the relay (the ports that get connected to the common) in port order.
27  The position value ranges from 0 to P-1, where 0 says that none of the throws are connected to
28  the common, and a value from 1 to P-1 indicates the throw port connected to the common.
29  Valid terminations are the string 'open', or a number indicating the impedance of the unconnected
30  port. This impedance is also applied to the common, when no ports are connected.
31  @param ports int number of ports in the relay (including the common).
32  @parem position int number from 0 to ports-1 indicating the relay position.
33  @param float or string (optional, defaults to 'open' termination applied to unconnected ports.
34  @param float (optional, defaults to 50) reference impedance.
35  @returns the list of list s-parameter matrix of the ideal relay in the position specified.
36  """
37  if termination == 'open': rho=1.
38  else: rho=TerminationZ(termination,Z0)[0][0]
39  S=[[rho if r==c else 0 for c in range(ports)] for r in range(ports)]
40  if 0 < position < ports:
41  S[position-1][position-1] = 0; S[position-1][ports-1] = 1.
42  S[ports-1][position-1] = 1.; S[ports-1][ports-1] = 0
43  return S
