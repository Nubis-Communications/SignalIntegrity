26 def S2Z(S,Z0=None,K=None):
27  """Converts s-parameters to Z-parameters
28  @param S list of list representing s-parameter matrix to convert
29  @param Z0 (optional) float or complex or list of reference impedance (defaults to None).
30  @param K (optional) float or complex or list of scaling factor (defaults to None).
31  @see Z0KHelper to see how the reference impedance
32  and scaling factor are determined."""
33  (Z0,K)=Z0KHelper((Z0,K),len(S))
34  I=identity(len(S))
35  S=array(S)
36  result = (K.dot(I+S).dot(inv(I-S)).dot(Z0).dot(inv(K))).tolist()
37  # pragma: silent exclude
38  if np.any(np.isnan(result)):
39  raise np.linalg.LinAlgError
40  # pragma: include
41  return result
