44 def VoltageAmplifierFourPort(G,Zi,Zo,Z0=50.):
45  """VoltageAmplifierFourPort
46  Four port voltage amplifier
47  @param G float voltage gain
48  @param Zi float or complex input impedance
49  @param Zo float or complex output impedance
50  @param Z0 (optional) float reference impedance (defaults to 50 ohms)
51  @return list of list s-parameter matrix for a voltage amplifier
52  @remark
53  The voltage sense element senses the voltage with the plus terminal at port 1
54  and and the minus terminal at port 2 shunted with impedance of Zi.\n
55  The voltage generated by the amplifier has the plus terminal at port 3 and
56  and the minus terminal at port 4 with Zo in series.\n
57  """
58  S11=Zi/(Zi+2.*Z0)
59  S12=2.*Z0/(Zi+2.*Z0)
60  S13=0
61  S14=0
62  S21=S12
63  S22=S11
64  S23=0
65  S24=0
66  S31=2.*Zi*Z0*G/((Zi+2.*Z0)*(Zo+2.*Z0))
67  S32=-S31
68  S33=Zo/(Zo+2.*Z0)
69  S34=2.*Z0/(Zo+2.*Z0)
70  S41=S32
71  S42=S31
72  S43=S34
73  S44=S33
74  return [[S11,S12,S13,S14],
75  [S21,S22,S23,S24],
76  [S31,S32,S33,S34],
77  [S41,S42,S43,S44]]

79 def VoltageAmplifierThreePort(G,Zi,Zo,Z0=50.):
80  """VoltageAmplifierThreePort
81  Three port voltage amplifier
82  @param G float voltage gain
83  @param Zi float or complex input impedance
84  @param Zo float or complex output impedance
85  @param Z0 (optional) float reference impedance (defaults to 50 ohms)
86  @return list of list s-parameter matrix for a voltage amplifier
87  @remark
88  The three port voltage amplifier is the same as the four port voltage amplifier with
89  ports two and three connected together and exposed as a single port.\n
90  The voltage sense element senses the voltage with the plus terminal at port 1
91  and and the minus terminal at port 3 shunted with impedance of Zi.\n
92  The voltage generated by the amplifier has the plus terminal at port 2 and
93  and the minus terminal at port 3 with Zo in series.\n
94  """
95  D=-Zo*Zi-2.*Zo*Z0-2.*Zi*Z0-3.*Z0*Z0+G*Zi*Z0
96  S11=(-Zo*Zi-2.*Zi*Z0+Z0*Z0+G*Zi*Z0)/D
97  S12=-2.*Z0*Z0/D
98  S13=-2.*Z0*(Zo+Z0)/D
99  S21=-2.*Z0*(G*Zi+Z0)/D
100  S22=(Z0*Z0-2.*Zo*Z0+G*Zi*Z0-Zo*Zi)/D
101  S23=2.*Z0*(G*Zi-Zi-Z0)/D
102  S31=2.*Z0*(-Z0+G*Zi-Zo)/D
103  S32=-2.*Z0*(Zi+Z0)/D
104  S33=(-Zo*Zi+Z0*Z0-G*Zi*Z0)/D
105  return [[S11,S12,S13],
106  [S21,S22,S23],
107  [S31,S32,S33]]

109 def VoltageAmplifierTwoPort(G,Zi,Zo,Z0=50.):
110  """VoltageAmplifierTwoPort
111  Two port voltage amplifier
112  @param G float voltage gain
113  @param Zi float or complex input impedance
114  @param Zo float or complex output impedance
115  @param Z0 (optional) float reference impedance (defaults to 50 ohms)
116  @return list of list s-parameter matrix for a voltage amplifier
117  @remark
118  The two port voltage amplifier is the same as the three port voltage amplifier with
119  port 3 grounded.\n
120  The voltage sense element senses the voltage to ground at port 1 shunted to ground
121  with impedance of Zi.\n
122  The voltage generated by the amplifier is at port 2 referenced to ground
123  with Zo in series with the output.\n
124  """
125  return [[(Zi-Z0)/(Zi+Z0),0.],[2.*G*Zi*Z0/((Zi+Z0)*(Zo+Z0)),(Zo-Z0)/(Zo+Z0)]]
