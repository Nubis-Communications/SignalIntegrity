184  def _ExtractionWindows(self,td,forwardSamples,reverseSamples,
185  raisedCosineSamples,maxValueIndex):

187  reflectExtractionWindow=Waveform(td)
188  for k in range(td.K):

190  incidentExtractionWindow[k]=0.0

192  elif k<maxValueIndex-reverseSamples and raisedCosineSamples>0:
