108  def ReadFromFile(self,filename):
109  """The file is read from the disk and the constants stored internally.
110  @param filename string name of calibration constant file to read.
111  @return self
112  """
113  with open(filename) as f:
114  lines=f.readlines()
115  actualLines=[]
116  for line in lines:
117  if line.strip()[0]!='%':
118  actualLines.append(line.strip())
119  # % C0 (fF) - OPEN
120  self.openC0=float(actualLines[0])*1e-15
121  # % C1 (1e-27 F/Hz) - OPEN
122  self.openC1=float(actualLines[1])*1e-27
123  # % C2 (1e-36 F/Hz^2) - OPEN
124  self.openC2=float(actualLines[2])*1e-36
125  # % C3 (1e-45 F/Hz^3) - OPEN
126  self.openC3=float(actualLines[3])*1e-45
127  # % offset delay (pS) - OPEN
128  self.openOffsetDelay=float(actualLines[4])*1e-12
129  # % real(Zo) of offset length - OPEN
130  self.openOffsetZ0=float(actualLines[5])
131  # % offset loss (Gohm/s) - OPEN
132  self.openOffsetLoss=float(actualLines[6])*1e9
133  # % L0 (pH) - SHORT
134  self.shortL0=float(actualLines[7])*1e-12
135  # % L1 (1e-24 H/Hz) - SHORT
136  self.shortL1=float(actualLines[8])*1e-24
137  # % L2 (1e-33 H/Hz^2) - SHORT
138  self.shortL2=float(actualLines[9])*1e-33
139  # % L3 (1e-42 H/Hz^3) - SHORT
140  self.shortL3=float(actualLines[10])*1e-42
141  # % offset delay (pS) - SHORT
142  self.shortOffsetDelay=float(actualLines[11])*1e-12
143  # % real(Zo) of offset length - SHORT

301  def __init__(self,filename=None,f=None):
302  """Constructor
303  @param filename (optional) string filename of calibration constants file to read.
304  @param f (optional) list of frequencies to define the calibration standards for.
305  The calibration constants are initialized to their default as specified in CalibrationConstants.__init__().
306

308 
309  If frequencies are provided, the frequencies are initialized and the calibration standards are initialized

311

313  @see IntializeFrequency
314  """
315  self.Constants=CalibrationConstants()
316  self.m_f=None
317  if not filename is None:
318  self.ReadFromFile(filename)
319  if not f is None:
320  self.InitializeFrequency(f)
321  def InitializeFrequency(self,f):
322  """Initializes frequencies
323  @param f list of frequencies
324  Calibration standards are initialized to have the s-parameters of the standards as defined by the
325  frequencies and calibration constants.
326 
327  Once intialized, the s-parameters of the calibration standards can be accessed and used.
328  @see openStandard
329  @see shortStandard
330  @see loadStandard
331  @see thruStandard
332  """
333  self.m_f=f
334  self.openStandard=OpenStandard(self.m_f,self.Constants.openOffsetDelay,
