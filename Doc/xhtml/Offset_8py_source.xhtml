32  """Constructor
33  @param fList list of frequencies
34  @param offsetDelay (optional) float electrical length of offset in s (defaults to 0 s)
35  @param offsetZ0 (optional) float real characteristic impedance of offset (defaults to 50 ohms)
36  @param offsetLoss (optional) float loss due to skin-effect defined in ohms/s at f0 (defaults to 0).
37  @param f0 (optional) float frequency where the offset loss is defined (defaults to 1e9).
38  The result is that the class becomes the base-class SParameters with the s-parameters
39  of an offset, which is a common portion of a calibration standard.
40  """
41  data=[]
42  Z0=50.
43  Zc=offsetZ0
44  Td=offsetDelay
45  R0=offsetLoss
46  L=Td*Zc
47  C=Td/Zc
48  G=0
49  for f in fList:
50  R=R0*Td*cmath.sqrt(f/f0)*(1+1j)
51  Z=R+1j*2*cmath.pi*f*L
52  Y=G+1j*2*cmath.pi*f*C
53  y=cmath.sqrt(Z*Y)
54  # pragma: silent exclude
55  try:
56  if self.calcZc:
57  Zc=cmath.sqrt(Z/Y)
58  except:
59  Zc=offsetZ0
60  # pragma: include
61  rho=(Zc-Z0)/(Zc+Z0)
62  D=(1-rho*rho*cmath.exp(-2*y))
63  S11=rho*(1-cmath.exp(-2*y))/D
64  S21=(1-rho*rho)*cmath.exp(-y)/D
65  data.append([[S11,S21],[S21,S11]])
