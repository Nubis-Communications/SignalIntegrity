24 class SParametersParser(SParameters):
25  """parses a list of commands to process s-parameters"""
26  def __init__(self,sp,lines):
27  """Constructor
28  Takes a set of s-parameters along with a list of command lines to process the s-parameters
29  @param sp instance of SParameters to process
30  @param lines list of lines with commands indicating the processing, in order
31  @remark For causality checks, the threshold for causality is 10e-6 (-100 dB).
32  The maximum number of iterations is 30.
33  the Largest singular value is 1.
34  @see EnforceCausality
35  @see EnforcePassivity
36  @see EnforceReciprocity
37  @see EnforceBothPassivityAndReciprocity
38  @see EnforceAll
39  @see LimitImpulseResponseLength
40  @see SetReferenceImpedance
41  @see RemoveImpulseResponseOffset
42  """
43  SParameters.__init__(self,sp.m_f,sp.m_d,sp.m_Z0)
44  for line in lines:
45  tokens=line.split()
46  if len(tokens)>=0:
47  if tokens[0]!='post': continue
48  else: # pragma: no cover
49  continue
50  try:
51  if tokens[1]=='enforce':
52  if tokens[2]=='causality':
53  self.EnforceCausality()
54  elif tokens[2]=='passivity':
55  self.EnforcePassivity()
56  elif tokens[2]=='reciprocity':
57  self.EnforceReciprocity()
58  elif tokens[2]=='both':
59  self.EnforceBothPassivityAndCausality(causalityThreshold=10e-6,maxIterations=30,maxSingularValue=1.)
60  elif tokens[2]=='all':
61  self.EnforceAll(causalityThreshold=10e-6,maxIterations=30,maxSingularValue=1.)
62  else:
63  raise IndexError
64  elif tokens[1]=='limit':
65  self.LimitImpulseResponseLength((float(tokens[2]) if tokens[2]!='none' else -1e15,
66  float(tokens[3]) if tokens[3]!='none' else +1e15))
67  elif tokens[1]=='offset':
68  if len(tokens)<= 2:
69  self.RemoveImpulseResponseOffset()
70  elif len(tokens)<=3:
71  self.RemoveImpulseResponseOffset((float(tokens[2]) if tokens[2]!='none' else -1e15,+1e15))
72  else:
73  self.RemoveImpulseResponseOffset((float(tokens[2]) if tokens[2]!='none' else -1e15,
74  float(tokens[3]) if tokens[3]!='none' else +1e15))
75  elif tokens[1]=='reference':
76  if tokens[2]=='impedance':
77  try:
78  self.SetReferenceImpedance(float(tokens[3]))
79  except:
80  raise IndexError
81  else:
82  raise IndexError
83  elif tokens[1][0]=='!':
84  self.header.append(line[len('post !'):])
85  else:
86  raise IndexError
87  except:
88  raise SignalIntegrityExceptionPostProcessing('not understood: '+line)
