147  def AddPort(self,DeviceName,DevicePort,SystemPort,AddThru=False):
148  """Adds a system port at a specified device port
149  @param DeviceName string name of device
150  @param DevicePort integer port number of device
151  @param SystemPort integer port number
152  @param AddThru boolean whether to force a thru between the system port and the device port.
153  Adds a system port to a device port for the solutions of s-parameters or for deembedding.
154 
155  In the theory, sometimes it's necessary to add a thru between a system port and a device port.
156  But it turns out that this is only necessary when connecting a system port directly to a port
157  of an unknown device in a deembedding problem. Although the ability to force the addition of
158  a thru is retained, the code actually checks for this situation and adds a thru automatically
159  so currently, there are no situations known where AddThru needs to be set True.
160 
161  @todo Consider removing the AddThru argument
162  """
163  PortName = 'P'+str(SystemPort)
164  self.AddDevice(PortName,1,[[0.0]])
165  self.AssignM(PortName,1,'m'+str(SystemPort))
166  if not AddThru:
167  AddThru = self[self.IndexOfDevice(DeviceName)].Type == 'unknown'
168  if AddThru:

170  self.AddDevice(thruName,2,Thru())
171  self.ConnectDevicePort(PortName,1,thruName,1)
172  self.ConnectDevicePort(thruName,2,DeviceName,DevicePort)
