210 class LogarithmicallySpacedFrequencyList(FrequencyList):
211  """a logarithmically spaced frequency list"""
212  def __init__(self,start_frequency,end_frequency,points_per_decade):
213  """Constructor
214  @param start_frequency float start frequency
215  @param end_frequency float end frequency
216  @param points_per_decade integer number of points per decade
217  Intitializes the frequency list with the list of frequencies
218 
219  The list of frequencies is constructed by first determining the list of frequencies in a decade based
220  on the number of points per decade specified. Then, for each decade, all of the points are retained
221  that are above the start frequency and below the end frequency. Finally, the start and end frequencies
222  are tacked onto the list.
223  """
224  import math
225  decade_list = [10.**(k/points_per_decade) for k in range(points_per_decade+1)]
226  frequency_list=[start_frequency]
227  for d in range(math.floor(math.log10(start_frequency)),math.ceil(math.log10(end_frequency))):
228  for k in range(points_per_decade+1):
229  this_frequency = decade_list[k] * math.pow(10.,d)
230  if (this_frequency > frequency_list[-1]) and (this_frequency < end_frequency):
231  frequency_list.append(this_frequency)
232  if end_frequency > frequency_list[-1]:
233  frequency_list.append(end_frequency)
234  FrequencyList.__init__(self,frequency_list)
