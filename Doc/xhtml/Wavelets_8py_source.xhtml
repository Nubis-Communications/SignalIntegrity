22 class Wavelet(object):
23  """base class that performs forward and inverse wavelet transforms
24  """
25  def __init__(self,h):
26  """constructs a wavelet transformer with the provided
27  wavelet coefficients.
28  @param h list of float wavelet coefficients
29  @remark The wavelet coefficients are generally supplied by a
30  derived class.
31  @see WaveletHaar
32  @see WaveletDaubechies4
33  @see WaveletDaubechies14
34  @see WaveletDaubechies16
35  """
36  self.h=h
37  self.L=len(self.h)
38  self.g=[pow(-1.,l)*h[self.L-1-l] for l in range(self.L)]
39  def DWT(self,xi):
40  """performs the discrete wavelet transform (DWT).
41  @param xi list of float input waveform values.
42  @warning length of xi must be a power of 2.
43  """
44  x=list(xi); N=len(x)
45  B=self.intlog2(N)-self.intlog2(self.L)+1
46  for i in range(B):
47  X=list(x)
48  for k in range(N//2):
49  X[k]=sum([x[(2*k+l)%N]*self.h[l]
50  for l in range(self.L)])
51  X[k+N//2]=sum([x[(2*k+l+N-2)%N]*self.g[l]
52  for l in range(self.L)])
53  x=list(X); N=N//2
54  return X
55  def IDWT(self,XI):
56  """peforms the inverse discrete wavelet transform (IDWT).
57  @param XI list of float wavelet transform.
58  @warning length of XI must be a power of 2.
59  """
60  X=list(XI); N=len(X)
61  B=self.intlog2(N)-self.intlog2(self.L)+1
62  N=self.L
63  for i in range(B):
64  x=list(X)
65  for k in range(N//2):
66  x[2*k]=sum([self.h[2*l]*X[(k-l+(N//2))%(N//2)]+
67  self.g[2*l]*X[(k+1-l+(N//2))%(N//2)+(N//2)]
68  for l in range(self.L//2)])
69  x[2*k+1]=sum([self.h[2*l+1]*X[(k-l+(N//2))%(N//2)]+
70  self.g[2*l+1]*X[(k+1-l+(N//2))%(N//2)+(N//2)]
71  for l in range(self.L//2)])
72  X=list(x); N=2*N
73  return x
74  @staticmethod
75  def intlog2(x):
76  """returns the closest power of 2 to x
77  @param x integer value
78  """
79  r=math.log(float(x))/math.log(2.)
80  return int(round(r))
