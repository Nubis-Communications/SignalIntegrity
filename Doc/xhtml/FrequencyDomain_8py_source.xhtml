43  """Values
44  @param unit (optional) string containing the unit for the frequencies
45  @return list of complex values corresponding to the frequency-domain elements in the
46  units specified.
47  @remark
48  Valid unit strings are:
49  - 'dB' - values in decibels.
50  - 'mag' - values in absolute magnitude.
51  - 'rad' - the argument or phase in radians.
52  - 'deg' - the argument or phase in degrees.
53  - 'real' - the real part of the values.
54  - 'imag' - the imaginary part of the values.
55 
56  Returns the list of complex values if no unit specified.
57 
58  Returns None if the unit is invalid.
59  """
60  if unit==None:
61  return list(self)
62  elif unit =='dB':
63  return [-3000. if (abs(self[n]) < 1e-15) else
64  20.*math.log10(abs(self[n]))
65  for n in range(len(self.m_f))]
66  elif unit == 'mag':
67  return [abs(self[n]) for n in range(len(self.m_f))]
68  elif unit == 'rad':
69  return [cmath.phase(self[n]) for n in range(len(self.m_f))]
70  elif unit == 'deg':
71  return [cmath.phase(self[n])*180./math.pi
72  for n in range(len(self.m_f))]
73  elif unit == 'real':
74  return [self[n].real for n in range(len(self.m_f))]
75  elif unit == 'imag':
