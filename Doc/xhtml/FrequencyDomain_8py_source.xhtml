53  def Values(self,unit=None):
54  """Values
55  @param unit (optional) string containing the unit for the frequencies
56  @return list of complex values corresponding to the frequency-domain elements in the
57  units specified.
58  @remark
59  Valid unit strings are:
60  - 'dB' - values in decibels.
61  - 'mag' - values in absolute magnitude.
62  - 'rad' - the argument or phase in radians.
63  - 'deg' - the argument or phase in degrees.
64  - 'real' - the real part of the values.
65  - 'imag' - the imaginary part of the values.
66 
67  Returns the list of complex values if no unit specified.
68 
69  Returns None if the unit is invalid.
70  """
71  if unit==None:
72  return list(self)
73  elif unit =='dB':
74  return [-3000. if (abs(self[n]) < 1e-15) else
75  20.*math.log10(abs(self[n]))
76  for n in range(len(self.m_f))]
77  elif unit == 'mag':
78  return [abs(self[n]) for n in range(len(self.m_f))]
79  elif unit == 'rad':
80  return [cmath.phase(self[n]) for n in range(len(self.m_f))]
81  elif unit == 'deg':
82  return [cmath.phase(self[n])*180./math.pi
83  for n in range(len(self.m_f))]
84  elif unit == 'real':
85  return [self[n].real for n in range(len(self.m_f))]
86  elif unit == 'imag':
87  return [self[n].imag for n in range(len(self.m_f))]
88  def ReadFromFile(self,fileName):
89  """reads in frequency domain content from the file specified.
90  @param fileName string file name to read
91  @return self
92  """
93  with open(fileName,'rU' if sys.version_info.major < 3 else 'r') as f:
94  data=f.readlines()
95  if data[0].strip('\n')!='UnevenlySpaced':
96  N = int(str(data[0]))
97  Fe = float(str(data[1]))
98  frl=[line.split(' ') for line in data[2:]]
99  resp=[float(fr[0])+1j*float(fr[1]) for fr in frl]
100  self.m_f=EvenlySpacedFrequencyList(Fe,N)
101  list.__init__(self,resp)
102  else:
103  frl=[line.split(' ') for line in data[1:]]
104  f=[float(fr[0]) for fr in frl]
105  resp=[float(fr[1])+1j*float(fr[2]) for fr in frl]
106  self.m_f=GenericFrequencyList(f)
107  list.__init__(self,resp)

146  def LimitEndFrequency(self,endFrequency):
147  """limits the end frequency
148  @param endFrequency float end frequency to limit to
149  @return self
150  @remark if the end frequency is higher than the current end frequency,
151  the content is left unchanged.
152  @warning the end frequency might be slightly higher and this is not
153  a strict limit. The goal is for the frequencies to potentially bracket
154  the desired end frequency.
155  """
156  frequencies=self.Frequencies()
157  deltaf=frequencies[-1]/(len(self)-1)
158  numPts=int(math.ceil(endFrequency/deltaf))
159  if numPts >= len(self):
160  return self

162  fl.CheckEvenlySpaced()
163  FrequencyDomain.__init__(self,fl,self[0:numPts+1])
164  return self
