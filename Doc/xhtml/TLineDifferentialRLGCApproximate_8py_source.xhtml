26 class TLineDifferentialRLGCApproximate(SParameters):
27  """s-parameters of differential RLGC (telegrapher's) transmission line
28  calculated by approximating distributed parameters with a finite number
29  of sections specified."""
30  rtFraction=.01

32  Cm, dfm, Gm, Lm, Z0=50., K=0, scale=1.):
33  """Constructor
34 
35  ports are 1,2,3,4 is +1,-1, +2, -2
36 
37  @param f list of float frequencies
38  @param Rp float DC resistance of positive leg (ohms)
39  @param Rsep float skin-effect resistance of positive leg (ohms/sqrt(Hz))
40  @param Lp float inductance of positive leg (H)
41  @param Gp float DC conductance of positive leg to ground (S)
42  @param Cp float capacitance of positive leg to ground (F)
43  @param dfp float dissipation factor (loss-tangent) of capacitance of positive leg to ground
44  @param Rn float DC resistance of negative leg (ohms)
45  @param Rsen float skin-effect resistance of negative leg (ohms/sqrt(Hz))
46  @param Ln float inductance of negative leg (H)
47  @param Gn float DC conductance of negative leg to ground (S)
48  @param Cn float capacitance of negative leg to ground (F)
49  @param dfn float dissipation factor (loss-tangent) of capacitance of negative leg to ground
50  @param Cm float mutual capacitance (F)
51  @param dfm float dissipation factor (loss-tangent) of mutual capacitance (F)
52  @param Gm float mutual conductance (S)
53  @param Lm float mutual inductance (H)
54  @param Z0 (optional) float reference impedance (defaults to 50 ohms)
55  @param K (optional) integer number of sections (defaults to 0)
56  @param scale (optional) float amount to scale the line by (assuming all other values are per unit)
57  @note If K=0 is specified, it is modified to a value that will provided a very good numerical
58  approximation.
59 
60  The calculation is such that round-trip propagation time (twice the electrical length)
61  of any one small section is no more than one percent of the fastest possible risetime.
62  """
63  Rp=Rp*scale; Rsep=Rsep*scale; Lp=Lp*scale; Gp=Gp*scale; Cp=Cp*scale; dfp=dfp
64  Rn=Rn*scale; Rsen=Rsen*scale; Ln=Ln*scale; Gn=Gn*scale; Cn=Cn*scale; dfn=dfn
65  Cm=Cm*scale; dfm=dfm; Gm=Gm*scale; Lm=Lm*scale
66  K=int(K*scale+0.5)
67  if K==0:
68  """max possible electrical length and fastest risetime"""
69  Td=math.sqrt((max(Lp,Ln)+Lm)*(max(Cp,Cn)+2*Cm)); Rt=0.45/f[-1]
70  """sections such that fraction of risetime less than round trip electrical
71  length of one section"""
72  K=int(math.ceil(Td*2/(Rt*self.rtFraction)))
73  self.m_K=K
