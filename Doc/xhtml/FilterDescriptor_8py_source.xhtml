22 class FilterDescriptor(object):
23  """handles time axis effects of filtering"""
24  def __init__(self,UpsampleFactor,DelaySamples,StartupSamples):
25  """Constructor
26  @param UpsampleFactor integer or float upsample factor. This is the multiplicative
27  factor that the filter will have on the sample rate of an applied waveform.
28  @param DelaySamples integer or float samples that the filter applies to a waveform.
29  @param StartupSamples integer or float the number of samples that must be removed from
30  the filtered waveform due to filter startup effects.
31  """
32  self.U = UpsampleFactor
33  self.D = DelaySamples
34  self.S = StartupSamples
35  def __eq__(self,other):
36  """overloads ==
37  @return boolean whether to filter descriptors are the same
38  """
39  if abs(self.U - other.U) > 1e-15: return False
40  if abs(self.D - other.D) > 1e-15: return False
41  if abs(self.S - other.S) > 1e-15: return False
42  return True
43  def __mul__(self,other):
44  """overloads *\n
45  multiplies two filter descriptors. This is the same as calculating the
46  effect of the two filters in cascade.
47  @param other an instance of class FilterDescriptor of another filter
48  @return instance of class FilterDescriptor of the cascaded filters
49  """
50  if isinstance(other,FilterDescriptor):
51  return FilterDescriptor(
52  UpsampleFactor=self.U*other.U,
53  DelaySamples=float(self.U*self.D+other.D)/self.U,
54  StartupSamples=float(self.U*self.S+other.S)/self.U)
55  def Before(self,other):
56  """calculates the effect of another filter before this one.
57  @param other an instance of class FilterDescriptor of another filter
58  @return instance of class FilterDescriptor of the other filter moved before this one
59  """
60  return FilterDescriptor(
61  UpsampleFactor=self.U,
62  DelaySamples=(other.D*other.U+self.D)/other.U-other.D/self.U,
63  StartupSamples=(other.U*other.S+self.S)/other.U-other.S/self.U)
64  def After(self,other):
65  """calculates the effect of another filter after this one.

67  @return instance of class FilterDescriptor of the other filter moved after this one
68  """
69  return FilterDescriptor(
70  UpsampleFactor=self.U,
71  DelaySamples=(self.D*self.U+other.D)*other.U/self.U-other.D*other.U,
72  StartupSamples=(self.S*self.U+other.S)*other.U/self.U-other.U*other.S)
