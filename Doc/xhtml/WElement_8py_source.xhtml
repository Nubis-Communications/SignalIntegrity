28  """A Maxwell matrix is a lower triangular matrix that is related to the mutual matrix."""

31  @param MM lower triangular list of list Maxwell matrix

34  def MutualMatrix(self):
35  """Mutual matrix
36  @return a lower triangular mutual matrix
37  """
38  wires=len(self)
39  Mm=[[None for _ in range(w+1)] for w in range(wires)]
40  for r in range(wires):
41  Mm[r][r]=sum([self[r][c] if r >= c else self[c][r] for c in range(wires)])
42  for c in range(r): Mm[r][c]=-self[r][c]
43  return MutualMatrix(Mm)

53  """Maxwell matrix

55  """
56  wires=len(self)
57  MM=[[None for _ in range(w+1)] for w in range(wires)]

63 class WElement(SParameters):
64  """s-parameters of of a W element is
65  calculated by approximating distributed parameters with a finite number
66  of sections specified."""
67  rtFraction=.01
68  def __init__(self,f, wires, R, Rse, df, Cm, Gm, Lm, Z0=50., K=0, scale=1.):
69  """Constructor
70  ports are 1,2,3,4, etc. on left and W, W+1, W+2, W+4, etc. on right where W=wires
71  and wire 1 is port 1 to P, wire 2 is port 2 to (P+1), etc. This is a P=2*w port device.
72  @param f list of float frequencies
73  @param wires integer number of wires
74  @param R list of float DC resistance of wire (ohms)
75  @param Rse list of float skin-effect resistance of wire (ohms/sqrt(Hz))
76  @param df list of list of float dissipation factor (loss-tangent) of capacitance (unitless)
77  @param Cm list of list of float mutual capacitance (self capacitance on diagonal) (F)
78  @param Gm list of list of float mutual conductance (self conductance on diagonal) (S)
79  @param Lm list of list of float mutual inductance (self inductance on diagonal) (H)
80  @param Z0 (optional) float reference impedance (defaults to 50 ohms)
81  @param K (optional) integer number of sections (defaults to 0)
82  @param scale (optional) float amount to scale the line by (assuming all other values are per unit)
83  @note If K=0 is specified, it is modified to a value that will provided a very good numerical
84  approximation.
85 
86  The calculation is such that round-trip propagation time (twice the electrical length)
87  of any one small section is no more than one percent of the fastest possible risetime.
88  """
89  R=[r*scale for r in R]; Rse=[rse*scale for rse in Rse]; Cm=[[cm*scale for cm in cmr] for cmr in Cm];
90  Gm=[[gm*scale for gm in gmr] for gmr in Gm]; Lm=[[lm*scale for lm in lmr] for lmr in Lm]
91  K=int(K*scale+0.5)
92  if K==0:
93  maxL=0.0; maxLm=0.0
94  for r in range(len(Lm)):
95  for c in range(len(Lm[r])):
96  if r==c: maxL=max(maxL,Lm[r][c])
97  else: maxLm=max(maxLm,Lm[r][c])
98  maxC=0.0; maxCm=0.0
99  for r in range(len(Cm)):
100  for c in range(len(Cm[r])):
101  if r==c: maxC=max(maxC,Cm[r][c])
102  else: maxCm=max(maxCm,Cm[r][c])
103  """max possible electrical length and fastest risetime"""
104  Td=math.sqrt((maxL+maxLm)*(maxC+2*maxCm)); Rt=0.45/f[-1]
105  """sections such that fraction of risetime less than round trip electrical
106  length of one section"""
107  K=int(math.ceil(Td*2/(Rt*self.rtFraction)))
108  self.m_K=K
109  # build the netlist
110  # pragma: silent exclude

113  from SignalIntegrity.Lib.Devices import TerminationG

121  sdp.AddLines(['device L_'+str(w+1)+' 2' for w in range(wires)])
122  sdp.AddLines(['device G_'+str(w+1)+' 1' for w in range(wires)])
123  sdp.AddLines(['device C_'+str(w+1)+' 1' for w in range(wires)])

125  for o in range(w+1,wires):
126  suffix='_'+str(w+1)+'_'+str(o+1)
127  sdp.AddLines(['device M'+suffix+' 4','device C'+suffix+' 2','device G'+suffix+' 2'])
128  sdp.AddLines(['port '+str(w+1)+' R_'+str(w+1)+' 1' for w in range(wires)])
129  sdp.AddLines(['port '+str(w+wires+1)+' C_'+str(w+1)+' 1' for w in range(wires)])

132  sdp.AddLines(['connect R_'+str(w+1)+' 2 Rse_'+str(w+1)+' 1' for w in range(wires)])
133  sdp.AddLines(['connect Rse_'+str(w+1)+' 2 L_'+str(w+1)+' 1' for w in range(wires)])
134  for w in range(wires):
135  DeviceFromString='L_'+str(w+1)
136  DeviceFromPin=2

138  for o in range(wires):

147  else:
148  actualFrom=o

150  actualInputPin=3
151  nextOutputPin=4
152  CapacitorConductanceConnectionString+=' C_'+str(o+1)+'_'+str(w+1)+' 2 G_'+str(o+1)+'_'+str(w+1)+' 2'
153

155  sdp.AddLine('connect '+DeviceFromString+' '+str(DeviceFromPin)+' '+DeviceToString+' '+str(actualInputPin))
156  DeviceFromString=DeviceToString
157  DeviceFromPin=nextOutputPin
158  # finally, connect this last device output to the capacitances and conductances
159  sdp.AddLine('connect '+DeviceFromString+' '+str(DeviceFromPin)+' '+CapacitorConductanceConnectionString)
160  # make the system description and assign the s-parameters
161  self.m_sspn=SystemSParametersNumeric(sdp.SystemDescription())
162  self.NetListLines=sdp.m_lines
163  # frequency independent get assigned directly
164  for w in range(wires):
165  self.m_sspn.AssignSParameters('R_'+str(w+1),SeriesZ(R[w]/K,Z0))
166  self.m_sspn.AssignSParameters('G_'+str(w+1),TerminationG(Gm[w][w]/K,Z0))
167  for o in range(w+1,wires):
168  self.m_sspn.AssignSParameters('G_'+str(w+1)+'_'+str(o+1),SeriesG(Gm[o][w]/K,Z0)) # [o][w] not [w][o] to allow lower triangular
169  # now for frequency dependent ones
170  self.m_spdl=[]
171  for w in range(wires):
172  self.m_spdl.append(('Rse_'+str(w+1),dev.SeriesRse(f,Rse[w]/K,Z0)))
173  self.m_spdl.append(('C_'+str(w+1),dev.TerminationC(f,Cm[w][w]/K,Z0,df[w][w])))
174  self.m_spdl.append(('L_'+str(w+1),dev.SeriesL(f,Lm[w][w]/K,Z0)))
175  for o in range(w+1,wires):
176  self.m_spdl.append(('M_'+str(w+1)+'_'+str(o+1),dev.Mutual(f,Lm[o][w]/K,Z0))) # see note about [o][w] order above
177  self.m_spdl.append(('C_'+str(w+1)+'_'+str(o+1),dev.SeriesC(f,Cm[o][w]/K,Z0,df[o][w])))
178  SParameters.__init__(self,f,None,Z0)
179  def NetList(self):
180  """NetList
181  @return the netlist formed for a W element.
182  @remark this is really for testing only.
183  """
184  return self.NetListLines
185  def __getitem__(self,n):
186  """overloads [n]
187  @return list of list s-parameter matrix for the nth frequency element
188  """
189  for ds in self.m_spdl: self.m_sspn.AssignSParameters(ds[0],ds[1][n])
190  sp=self.m_sspn.SParameters()
191  if sp == 1: return sp
192  lp=[w+1 for w in range(len(sp)//2)]; rp=[w+len(sp)//2+1 for w in range(len(sp)//2)]
193  return T2S(linalg.matrix_power(S2T(sp,lp,rp),self.m_K),lp,rp)

196  """W element file."""
197  def __init__(self,f,filename,df=0.,Z0=50., K=0, scale=1.):
198  """Constructor
199  @param f list of frequencies
200  @param filename string W element file.
201  @param df float (optional, defaults to 0.) dissipation factor to be added to W elements.
202  @param Z0 float (optional, defaults to 50.) reference impedance.
203  @param K int (optional, defaults to 0.) number of sections to be used for approximation.
204  @param scale float (optional, defaults to 1.) scale factor to scale result by.
205  @remark dissipation factor does not seem to be included in W elements, which is surprising, which
206  neccessitates the addition through an argument until I better understand this.
207  @remark If sections are specified as 0, the number of sections used for the approximation will be determined
208  automatically based on capacitances and inductances supplied, and the final end frequency for the s-parameters.
209  """
210  self.Z0=Z0
211  self.K=K
212  self.scale=scale
213  with open(filename,'rU' if sys.version_info.major < 3 else 'r') as fi:
214  lines=fi.readlines()
215  self.numbersList=[]
216  for line in lines:
217  if len(line)==0: continue
218  elif line[0]=='*': continue
219  elif len(line.strip(' '))==0: continue
220  tokens=line.split()
221  if len(tokens)>0:
222  self.numbersList.extend(tokens)
223  self.idx=0
224  self.wires=int(self._ReadToken())
225  self.L0=[[float(self._ReadToken()) for _ in range(w+1)] for w in range(self.wires)]
226  CM=[[float(self._ReadToken()) for _ in range(w+1)] for w in range(self.wires)]
227  # CM is a Maxwell capacitance matrix. It needs to be converted to a mutual capacitance matrix
228  self.C0=MaxwellMatrix(CM).MutualMatrix()
229  self.R0=[[float(self._ReadToken()) for _ in range(w+1)] for w in range(self.wires)]
230  self.G0=[[float(self._ReadToken()) for _ in range(w+1)] for w in range(self.wires)]
231  self.Rs=[[float(self._ReadToken()) for _ in range(w+1)] for w in range(self.wires)]
232  self.Gd=[[float(self._ReadToken()) for _ in range(w+1)] for w in range(self.wires)]
233  self.df=[[df for _ in range(w+1)] for w in range(self.wires)]
234  WElement.__init__(self,f,self.wires,[self.R0[w][w] for w in range(self.wires)],[self.Rs[w][w] for w in range(self.wires)],
235  self.df,self.C0,self.G0,self.L0,self.Z0,self.K,self.scale)
236  def _ReadToken(self):
237  """Reads a single numeric token in string form from the W element file.
238  @return string numeric token.
239  """
